### Examples

Let's see how to use the `Menu` components. Let's start with a basic example:

```tsx
import {
  MenuRoot,
  MenuTrigger,
  MenuPositioner,
  MenuContent,
  MenuItem
} from "@/components/ui/menu"
import { Button } from "@/components/ui/button"

export const BasicMenu = () => (
  <MenuRoot>
    <MenuTrigger asChild>
      <Button variant="outline">Open Menu</Button>
    </MenuTrigger>

    <MenuPositioner>
      <MenuContent>
        <MenuItem value="copy">Copy</MenuItem>

        <MenuItem value="paste">Paste</MenuItem>

        <MenuItem value="cut">Cut</MenuItem>
      </MenuContent>
    </MenuPositioner>
  </MenuRoot>
)
```

You can organize menu items into groups:

```tsx
import {
  MenuRoot,
  MenuTrigger,
  MenuPositioner,
  MenuContent,
  MenuItem,
  MenuItemGroup,
  MenuItemGroupLabel,
  MenuSeparator
} from "@/components/ui/menu"
import { Button } from "@/components/ui/button"

export const MenuWithGroups = () => (
  <MenuRoot>
    <MenuTrigger asChild>
      <Button>Menu with Groups</Button>
    </MenuTrigger>

    <MenuPositioner>
      <MenuContent>
        <MenuItemGroup>
          <MenuItemGroupLabel>Actions</MenuItemGroupLabel>

          <MenuItem value="copy">Copy</MenuItem>

          <MenuItem value="paste">Paste</MenuItem>

          <MenuItem value="cut">Cut</MenuItem>
        </MenuItemGroup>

        <MenuSeparator />

        <MenuItemGroup>
          <MenuItemGroupLabel>View</MenuItemGroupLabel>

          <MenuItem value="zoom-in">Zoom In</MenuItem>

          <MenuItem value="zoom-out">Zoom Out</MenuItem>

          <MenuItem value="reset" disabled>Reset View</MenuItem>
        </MenuItemGroup>
      </MenuContent>
    </MenuPositioner>
  </MenuRoot>
)
```
