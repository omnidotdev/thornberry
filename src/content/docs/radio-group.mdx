---
title: Radio Group
description: A set of checkable buttons—known as radio buttons—where no more than one of the buttons can be checked at a time.
---

## Usage

```tsx
import {
  RadioGroupRoot,
  RadioGroupLabel,
  RadioGroupIndicator,
  RadioGroupItem,
  RadioGroupItemText,
  RadioGroupItemControl,
  RadioGroupItemHiddenInput
} from "@/components/ui/radio-group";
```

```tsx
<RadioGroupRoot defaultValue="option1">
  <RadioGroupLabel>Choose an option</RadioGroupLabel>
  <RadioGroupItem value="option1">
    <RadioGroupItemHiddenInput />
    <RadioGroupItemControl />
    <RadioGroupItemText>Option 1</RadioGroupItemText>
  </RadioGroupItem>
  <RadioGroupItem value="option2">
    <RadioGroupItemHiddenInput />
    <RadioGroupItemControl />
    <RadioGroupItemText>Option 2</RadioGroupItemText>
  </RadioGroupItem>
  <RadioGroupItem value="option3">
    <RadioGroupItemHiddenInput />
    <RadioGroupItemControl />
    <RadioGroupItemText>Option 3</RadioGroupItemText>
  </RadioGroupItem>
</RadioGroupRoot>
```
