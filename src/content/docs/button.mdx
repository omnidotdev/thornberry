## Installation

<Tabs items={["bun", "pnpm", "yarn", "npm"]} className="mt-4">

    ```
    bunx --bun shadcn@latest add http://localhost:3000/r/button.json
    ```
  </Tab>
  <Tab value="pnpm">
    ```
    pnpm dlx shadcn@latest add http://localhost:3000/r/button.json
    ```
  </Tab>
  <Tab value="yarn">
    ```
    npx shadcn@latest add http://localhost:3000/r/button.json
    ```
  </Tab>
  <Tab value="npm">
    ```
    npx shadcn@latest add http://localhost:3000/r/button.json
    ```
  </Tab>
</Tabs>

```tsx
import { Button } from "@/components/ui/button";
```

```tsx
<Button>Click me</Button>
```
