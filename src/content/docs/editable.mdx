## Usage

```tsx
import {
  EditableRoot,
  EditableLabel,
  EditableArea,
  EditableInput,
  EditablePreview,
  EditableControl,
  EditableEditTrigger,
  EditableSubmitTrigger,
  EditableCancelTrigger
} from "@/components/ui/editable";
```

```tsx
<EditableRoot>
  <EditableLabel>Label</EditableLabel>
  <EditableArea>
    <EditableInput />
    <EditablePreview />
  </EditableArea>
  <EditableControl>
    <EditableEditTrigger />
    <EditableSubmitTrigger />
    <EditableCancelTrigger />
  </EditableControl>
</EditableRoot>
```
