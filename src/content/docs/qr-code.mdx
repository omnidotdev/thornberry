import { QrCodeRoot, QrCodeFrame, QrCodePattern, QrCodeDownloadTrigger } from "@/registry/thornberry/components/qr-code";

<QrCodeRoot defaultValue="https://thornberry.omnidev.io">
  <QrCodeFrame>
    <QrCodePattern />
  </QrCodeFrame>
  <QrCodeDownloadTrigger fileName="qr-code.png" mimeType="image/png">
    Download QR Code
  </QrCodeDownloadTrigger>
</QrCodeRoot>

## Usage

```tsx
import {
  QrCodeRoot,
  QrCodeFrame,
  QrCodePattern,
  QrCodeDownloadTrigger,
} from "@/components/ui/qr-code";
```

```tsx
const Example = () => (
  <QrCodeRoot defaultValue="https://example.com">
    <QrCodeFrame>
      <QrCodePattern />
    </QrCodeFrame>
    <QrCodeDownloadTrigger fileName="qr-code.png" mimeType="image/png">
      Download
    </QrCodeDownloadTrigger>
  </QrCodeRoot>
);
```
