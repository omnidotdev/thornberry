### Examples

Let's see how to use the `Tabs` components. Let's start with a basic example:

```tsx
import {
  TabsRoot,
  TabsList,
  TabsTrigger,
  TabsContent
} from "@/components/ui/tabs"

export const BasicTabs = () => (
  <TabsRoot defaultValue="account">
    <TabsList>
      <TabsTrigger value="account">Account</TabsTrigger>
      
      <TabsTrigger value="password">Password</TabsTrigger>
    </TabsList>

    <TabsContent value="account">
      Manage your account settings and preferences.
    </TabsContent>

    <TabsContent value="password">
      Change your password and security settings.
    </TabsContent>
  </TabsRoot>
)
```

```tsx
import {
  TabsRoot,
  TabsList,
  TabsTrigger,
  TabsContent,
  TabsIndicator
} from "@/components/ui/tabs"

export const CustomTabs = () => (
  <TabsRoot defaultValue="primitives">
    <TabsList>
      <TabsTrigger value="primitives">Primitives</TabsTrigger>

      <TabsIndicator />
    </TabsList>

    <TabsContent value="primitives">
      Tabs are built using primitive components that you can compose to create custom tab interfaces.
    </TabsContent>

    <TabsContent value="usage">
      Usage content here
    </TabsContent>

    <TabsContent value="styling">
      Styling content here
    </TabsContent>
  </TabsRoot>
)
```
