### Examples

Let's see how to use the `Tabs` component. Let's start with a basic example:

```tsx
import { Tabs } from "@/components/ui/tabs"

export const BasicTabs = () => (
  <Tabs
    tabs={[
      {
        value: "account",
        trigger: "Account",
        content: (
          <div className="p-4 rounded-md border">
            <h3 className="text-lg font-medium">Account Settings</h3>
            <p className="text-sm text-muted-foreground mt-2">
              Manage your account settings and preferences.
            </p>
          </div>
        )
      },
      {
        value: "password",
        trigger: "Password",
        content: (
          <div className="p-4 rounded-md border">
            <h3 className="text-lg font-medium">Password Settings</h3>
            <p className="text-sm text-muted-foreground mt-2">
              Change your password and security settings.
            </p>
          </div>
        )
      }
    ]}
  />
)
```

For more granular control, Thornberry provides Tabs primitives:

```tsx
import {
  TabsRoot,
  TabsList,
  TabsTrigger,
  TabsContent
} from "@/components/ui/tabs"

export const TabsPrimitives = () => (
  <TabsRoot defaultValue="primitives">
    <TabsList>
      <TabsTrigger value="primitives">Primitives</TabsTrigger>
      <TabsTrigger value="usage">Usage</TabsTrigger>
      <TabsTrigger value="styling">Styling</TabsTrigger>
    </TabsList>
    <TabsContent value="primitives">
      <div className="p-4 rounded-md border">
        <div className="text-sm">
          Tabs are built using primitive components that you can compose to create custom tab interfaces.
        </div>
      </div>
    </TabsContent>
    <TabsContent value="usage">
      <div className="p-4 rounded-md border">
        <div className="text-sm">
          Usage content here
        </div>
      </div>
    </TabsContent>
    <TabsContent value="styling">
      <div className="p-4 rounded-md border">
        <div className="text-sm">
          Styling content here
        </div>
      </div>
    </TabsContent>
  </TabsRoot>
)
```
