---
title: Popover
description: A small floating content display that appears when a user interacts with a trigger element
---

import {
  Popover,
  PopoverArrow,
  PopoverArrowTip,
  PopoverCloseTrigger,
  PopoverContent,
  PopoverContext,
  PopoverPositioner,
  PopoverProvider,
  PopoverRoot,
  PopoverTrigger,
} from "@/registry/thornberry/components/popover";
import { Button } from "@/registry/thornberry/components/button";
import { Input } from "@/registry/thornberry/components/input";

## Basic

```tsx
<Popover trigger={<Button>Open</Button>}>
  <div className="space-y-2">
    <h4 className="font-medium leading-none">Dimensions</h4>
    <div className="text-sm text-muted-foreground">Set the dimensions for the object.</div>
    <div className="flex gap-2">
      <div className="grid gap-1">
        <label htmlFor="width" className="text-xs">Width</label>
        <Input id="width" defaultValue="100%" />
      </div>
      <div className="grid gap-1">
        <label htmlFor="height" className="text-xs">Height</label>
        <Input id="height" defaultValue="25px" />
      </div>
    </div>
  </div>
</Popover>
```

## With Close Trigger

```tsx
<Popover trigger={<Button>Open</Button>}>
  <div className="space-y-3">
    <div className="flex items-center justify-between">
      <h4 className="font-medium leading-none">Notifications</h4>
      <PopoverCloseTrigger />
    </div>
    <div className="text-sm text-muted-foreground">
      You have 3 unread messages.
    </div>
    <div className="pt-2 border-t">
      <Button size="sm" className="w-full">View All</Button>
    </div>
  </div>
</Popover>
```

## Positioning

```tsx
<Popover
  trigger={<Button variant="outline">Left</Button>}
  positioning={{ placement: "left" }}
>
  This popover appears on the left
</Popover>
```

## Primitives

```tsx
<PopoverRoot>
  <PopoverTrigger asChild>
    <Button variant="outline">Open</Button>
  </PopoverTrigger>
  <PopoverPositioner>
    <PopoverContent>
      <PopoverArrow>
        <PopoverArrowTip />
      </PopoverArrow>
      <div className="space-y-2">
        <h4 className="font-medium">Custom Styled Popover</h4>
        <div className="text-sm">
          This popover uses primitive components for custom styling.
        </div>
        <div className="flex justify-end">
          <PopoverCloseTrigger asChild>
            <Button variant="outline" size="sm">Close</Button>
          </PopoverCloseTrigger>
        </div>
      </div>
    </PopoverContent>
  </PopoverPositioner>
</PopoverRoot>
```
