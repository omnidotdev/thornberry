title: Popover
description: A small floating content display that appears when a user interacts with a trigger element.
---

import { PopoverRoot, PopoverTrigger, PopoverPositioner, PopoverContent, PopoverCloseTrigger } from "@/registry/thornberry/components/popover";
import { Button } from "@/registry/thornberry/components/button";
import { Input } from "@/registry/thornberry/components/input";

<PopoverRoot>
  <PopoverTrigger asChild>
    <Button className="mb-8">Open</Button>
  </PopoverTrigger>
  <PopoverPositioner>
    <PopoverContent>
      <div className="space-y-2">
        <h4 className="font-medium leading-none">Dimensions</h4>
        <div className="text-sm text-muted-foreground">Set the dimensions for the object.</div>
        <div className="flex gap-2">
          <div className="grid gap-1">
            <label htmlFor="width" className="text-xs">Width</label>
            <Input id="width" defaultValue="100%" className="h-8" />
          </div>
          <div className="grid gap-1">
            <label htmlFor="height" className="text-xs">Height</label>
            <Input id="height" defaultValue="25px" className="h-8" />
          </div>
        </div>
      </div>
    </PopoverContent>
  </PopoverPositioner>
</PopoverRoot>

### Examples

Let's see how to use the `Popover` components. Let's start with a basic example:

```tsx
import { 
  PopoverRoot, 
  PopoverTrigger, 
  PopoverPositioner, 
  PopoverContent 
} from "@/components/ui/popover"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"

export const BasicPopover = () => (
  <PopoverRoot>
    <PopoverTrigger asChild>
      <Button>Open</Button>
    </PopoverTrigger>
    <PopoverPositioner>
      <PopoverContent>
        <div className="space-y-2">
          <h4 className="font-medium leading-none">Dimensions</h4>
          <div className="text-sm text-muted-foreground">Set the dimensions for the object.</div>
          <div className="flex gap-2">
            <div className="grid gap-1">
              <label htmlFor="width" className="text-xs">Width</label>
              <Input id="width" defaultValue="100%" />
            </div>
            <div className="grid gap-1">
              <label htmlFor="height" className="text-xs">Height</label>
              <Input id="height" defaultValue="25px" />
            </div>
          </div>
        </div>
      </PopoverContent>
    </PopoverPositioner>
  </PopoverRoot>
)
```

You can add a close trigger to the popover content:

```tsx
import { 
  PopoverRoot, 
  PopoverTrigger, 
  PopoverPositioner, 
  PopoverContent, 
  PopoverCloseTrigger 
} from "@/components/ui/popover"
import { Button } from "@/components/ui/button"

export const PopoverWithClose = () => (
  <PopoverRoot>
    <PopoverTrigger asChild>
      <Button>Open</Button>
    </PopoverTrigger>
    <PopoverPositioner>
      <PopoverContent>
        <div className="space-y-3">
          <div className="flex items-center justify-between">
            <h4 className="font-medium leading-none">Notifications</h4>
            <PopoverCloseTrigger />
          </div>
          <div className="text-sm text-muted-foreground">
            You have 3 unread messages.
          </div>
          <div className="pt-2 border-t">
            <Button size="sm" className="w-full">View All</Button>
          </div>
        </div>
      </PopoverContent>
    </PopoverPositioner>
  </PopoverRoot>
)
```

You can customize the positioning of the popover:

```tsx
import { 
  PopoverRoot, 
  PopoverTrigger, 
  PopoverPositioner, 
  PopoverContent 
} from "@/components/ui/popover"
import { Button } from "@/components/ui/button"

export const PositionedPopovers = () => (
  <div className="flex gap-4">
    <PopoverRoot>
      <PopoverTrigger asChild>
        <Button variant="outline">Left</Button>
      </PopoverTrigger>
      <PopoverPositioner placement="left">
        <PopoverContent>
          This popover appears on the left
        </PopoverContent>
      </PopoverPositioner>
    </PopoverRoot>
    
    <PopoverRoot>
      <PopoverTrigger asChild>
        <Button variant="outline">Top</Button>
      </PopoverTrigger>
      <PopoverPositioner placement="top">
        <PopoverContent>
          This popover appears on top
        </PopoverContent>
      </PopoverPositioner>
    </PopoverRoot>
    
    <PopoverRoot>
      <PopoverTrigger asChild>
        <Button variant="outline">Bottom</Button>
      </PopoverTrigger>
      <PopoverPositioner placement="bottom">
        <PopoverContent>
          This popover appears on the bottom
        </PopoverContent>
      </PopoverPositioner>
    </PopoverRoot>
    
    <PopoverRoot>
      <PopoverTrigger asChild>
        <Button variant="outline">Right</Button>
      </PopoverTrigger>
      <PopoverPositioner placement="right">
        <PopoverContent>
          This popover appears on the right
        </PopoverContent>
      </PopoverPositioner>
    </PopoverRoot>
  </div>
)
```

```tsx
import {

          <div className="text-sm">
            This popover includes an arrow and custom styling.
          </div>
          <div className="flex justify-end">
            <PopoverCloseTrigger asChild>
              <Button variant="outline" size="sm">Close</Button>
            </PopoverCloseTrigger>
          </div>
        </div>
      </PopoverContent>
    </PopoverPositioner>
  </PopoverRoot>
)
```
