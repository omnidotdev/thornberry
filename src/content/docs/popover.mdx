---
title: Popover
description: A small floating content display that appears when a user interacts with a trigger element.
---

import { FiX, FiInfo, FiSettings, FiUser } from "react-icons/fi";
import { PopoverRoot, PopoverTrigger, PopoverPositioner, PopoverContent, PopoverCloseTrigger, PopoverArrow, PopoverArrowTip } from "@/registry/thornberry/components/popover";
import { Button } from "@/registry/thornberry/components/button";
import { Input } from "@/registry/thornberry/components/input";

<PopoverRoot>
  <PopoverTrigger asChild className="mb-8">
    <Button>Open Popover</Button>
  </PopoverTrigger>
  <PopoverPositioner>
    <PopoverContent>
      <div className="space-y-2">
        <h4 className="font-medium leading-none">Dimensions</h4>
        <div className="text-sm text-muted-foreground">Set the dimensions for the object.</div>
        <div className="flex gap-2">
          <div className="grid gap-1">
            <label htmlFor="width" className="text-xs">Width</label>
            <Input id="width" defaultValue="100%" className="h-8" />
          </div>
          <div className="grid gap-1">
            <label htmlFor="height" className="text-xs">Height</label>
            <Input id="height" defaultValue="25px" className="h-8" />
          </div>
        </div>
      </div>
    </PopoverContent>
  </PopoverPositioner>
</PopoverRoot>

## Usage

```tsx
import {
  PopoverRoot,
  PopoverTrigger,
  PopoverPositioner,
  PopoverContent
} from "@/components/ui/popover";
import { Button } from "@/components/ui/button";
```

```tsx
<PopoverRoot>
  <PopoverTrigger asChild>
    <Button>Open Popover</Button>
  </PopoverTrigger>
  <PopoverPositioner>
    <PopoverContent>
      <h4 className="font-medium">Title</h4>
      <p>Popover content goes here.</p>
    </PopoverContent>
  </PopoverPositioner>
</PopoverRoot>
```
