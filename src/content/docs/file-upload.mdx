## Usage

```tsx
import {
  FileUpload,
  FileUploadDropzone,
  FileUploadTrigger,
  FileUploadItemGroup,
  FileUploadItem,
  FileUploadItemName,
  FileUploadItemSizeText,
  FileUploadItemDeleteTrigger,
  FileUploadLabel,
  FileUploadHiddenInput,
  FileUploadContext,
} from "@/components/ui/file-upload";
```

```tsx
<FileUpload maxFiles={3}>
  <FileUploadLabel>Upload files</FileUploadLabel>
  <FileUploadDropzone>
    <p>Drag and drop files here or</p>
    <FileUploadTrigger>Choose files</FileUploadTrigger>
  </FileUploadDropzone>
  <FileUploadItemGroup>
    <FileUploadContext>
      {({ acceptedFiles }) =>
        acceptedFiles.map((file) => (
          <FileUploadItem key={file.name} file={file}>
            <div>
              <FileUploadItemName />
              <FileUploadItemSizeText />
            </div>
            <FileUploadItemDeleteTrigger>
             x
            </FileUploadItemDeleteTrigger>
          </FileUploadItem>
        ))
      }
    </FileUploadContext>
  </FileUploadItemGroup>
  <FileUploadHiddenInput />
</FileUpload>
```
