import { AccordionRoot, AccordionItem, AccordionItemTrigger, AccordionItemIndicator, AccordionItemContent } from "@/registry/thornberry/components/accordion";

<AccordionRoot
  multiple
  defaultValue={['React']}
  className="mb-8"
>
  <AccordionItem value="React">
    <AccordionItemTrigger>React</AccordionItemTrigger>
    <AccordionItemContent>this is the content for React</AccordionItemContent>
  </AccordionItem>

  <AccordionItem value="Vue">
    <AccordionItemTrigger>Vue</AccordionItemTrigger>
    <AccordionItemContent>this is the content for Vue</AccordionItemContent>
  </AccordionItem>

  <AccordionItem value="Solid" disabled>
    <AccordionItemTrigger>Solid</AccordionItemTrigger>
    <AccordionItemContent>this is the content for Solid</AccordionItemContent>
  </AccordionItem>
</AccordionRoot>
