---
title: Collapsible
description: An interactive component that can be opened and closed to reveal or hide content
---

import {
  Collapsible,
  CollapsibleContent,
  CollapsibleContext,
  CollapsibleProvider,
  CollapsibleRoot,
  CollapsibleTrigger,
} from "@/registry/thornberry/components/collapsible";

## Basic

```tsx
<Collapsible trigger="Click to expand">
  <div className="p-4 pt-2">
    <p>This content can be expanded and collapsed with the click of a button.</p>
  </div>
</Collapsible>
```

<Collapsible trigger="Click to expand" className="my-4">
  <div className="p-4 pt-2">
    <p>This content can be expanded and collapsed with the click of a button.</p>
  </div>
</Collapsible>

## Custom Styling

```tsx
<Collapsible
  trigger={
    <div className="flex items-center gap-2">
      <div className="h-3 w-3 rounded-full bg-blue-500" />
      <span>Custom trigger with icon</span>
    </div>
  }
>
  <div className="rounded-b-md border border-t-0 p-4 shadow-sm">
    <p>You can style the collapsible components to match your design.</p>
  </div>
</Collapsible>
```

## Primitives

```tsx
<CollapsibleRoot>
  <CollapsibleTrigger>
    Click to reveal primitives example
  </CollapsibleTrigger>
  <CollapsibleContent>
    <p>This example uses the primitive components to have more control over the structure.</p>
    <p className="mt-2">You can customize each part independently.</p>
  </CollapsibleContent>
</CollapsibleRoot>
```

<CollapsibleRoot className="my-4">
  <CollapsibleTrigger className="bg-purple-100 hover:bg-purple-200 data-[state=open]:bg-purple-200">
    Click to reveal primitives example
  </CollapsibleTrigger>
  <CollapsibleContent className="border border-t-0 p-4 rounded-b-md bg-purple-50">
    <p>This example uses the primitive components to have more control over the structure.</p>
    <p className="mt-2">You can customize each part independently.</p>
  </CollapsibleContent>
</CollapsibleRoot>
