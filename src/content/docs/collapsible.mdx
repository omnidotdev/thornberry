## Installation

<Tabs items={["bun", "pnpm", "yarn", "npm"]} className="mt-4">
  <Tab value="bun">

  </Tab>
  <Tab value="pnpm">
    ```
    pnpm dlx shadcn@latest add http://localhost:3000/r/collapsible.json
    ```
  </Tab>
  <Tab value="yarn">
    ```
    npx shadcn@latest add http://localhost:3000/r/collapsible.json
    ```
  </Tab>
  <Tab value="npm">
    ```
    npx shadcn@latest add http://localhost:3000/r/collapsible.json
    ```
  </Tab>
</Tabs>

## Usage

```tsx
import { CollapsibleRoot, CollapsibleTrigger, CollapsibleContent } from "@/components/ui/collapsible";
```

```tsx
<CollapsibleRoot>
  <CollapsibleTrigger>Click to expand</CollapsibleTrigger>
  <CollapsibleContent>
    This content can be expanded and collapsed.
  </CollapsibleContent>
</CollapsibleRoot>
```
