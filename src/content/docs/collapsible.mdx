### Examples

Let's see how to use the `Collapsible` components. Let's start with a basic example:

```tsx
import { CollapsibleRoot, CollapsibleTrigger, CollapsibleContent } from "@/components/ui/collapsible"

export const BasicCollapsible = () => (
  <CollapsibleRoot>
    <CollapsibleTrigger>Click to expand</CollapsibleTrigger>
    
    <CollapsibleContent>
      This content can be expanded and collapsed with the click of a button.
    </CollapsibleContent>
  </CollapsibleRoot>
)
```

You can customize the trigger with your own content:

```tsx
import { CollapsibleRoot, CollapsibleTrigger, CollapsibleContent } from "@/components/ui/collapsible"

export const CustomTriggerCollapsible = () => (
  <CollapsibleRoot>
    <CollapsibleTrigger>
      Custom trigger with icon
    </CollapsibleTrigger>
    
    <CollapsibleContent>
      You can style the collapsible components to match your design.
    </CollapsibleContent>
  </CollapsibleRoot>
)
```

```tsx
import { CollapsibleRoot, CollapsibleTrigger, CollapsibleContent } from "@/components/ui/collapsible"

export const AnimatedCollapsible = () => (
  <CollapsibleRoot>
    <CollapsibleTrigger>
      Click for animated content
    </CollapsibleTrigger>
    
    <CollapsibleContent>
      This example uses custom animations for the collapsible content.
      You can customize each part independently.
    </CollapsibleContent>
  </CollapsibleRoot>
)
```
