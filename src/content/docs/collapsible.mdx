import { Settings, ChevronRight } from "lucide-react";
import { CollapsibleRoot, CollapsibleTrigger, CollapsibleContent } from "@/registry/thornberry/components/collapsible";

<CollapsibleRoot className="mb-8">
  <CollapsibleTrigger>
      Click to expand
      <ChevronRight className="h-4 w-4 transition-transform" />
  </CollapsibleTrigger>
  <CollapsibleContent>
    <p>This content can be expanded and collapsed with the click of a button.</p>
  </CollapsibleContent>
</CollapsibleRoot>

## Usage

```tsx
import { CollapsibleRoot, CollapsibleTrigger, CollapsibleContent } from "@/components/ui/collapsible";
```

```tsx
<CollapsibleRoot>
  <CollapsibleTrigger>
    Click to expand
    <ChevronRight className="h-4 w-4 transition-transform" />  
  </CollapsibleTrigger>
  <CollapsibleContent>
    This content can be expanded and collapsed.
  </CollapsibleContent>
</CollapsibleRoot>
```
