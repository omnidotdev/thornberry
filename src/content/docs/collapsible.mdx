### Examples

Let's see how to use the `Collapsible` component. Let's start with a basic example:

```tsx
import { Collapsible } from "@/components/ui/collapsible"

export const BasicCollapsible = () => (
  <Collapsible trigger="Click to expand">
    <div className="p-4 pt-2">
      <p>This content can be expanded and collapsed with the click of a button.</p>
    </div>
  </Collapsible>
)
```

You can customize the trigger with your own content:

```tsx
import { Collapsible } from "@/components/ui/collapsible"

export const CustomTriggerCollapsible = () => (
  <Collapsible
    trigger={
      <div className="flex items-center gap-2">
        <div className="h-3 w-3 rounded-full bg-blue-500" />
        <span>Custom trigger with icon</span>
      </div>
    }
  >
    <div className="rounded-b-md border border-t-0 p-4 shadow-sm">
      <p>You can style the collapsible components to match your design.</p>
    </div>
  </Collapsible>
)
```

For more granular control, Thornberry provides Collapsible primitives:

```tsx
import {
  CollapsibleRoot,
  CollapsibleTrigger,
  CollapsibleContent
} from "@/components/ui/collapsible"

export const CollapsiblePrimitives = () => (
  <CollapsibleRoot>
    <CollapsibleTrigger>
      Click to reveal primitives example
    </CollapsibleTrigger>
    <CollapsibleContent>
      <p>This example uses the primitive components to have more control over the structure.</p>
      <p className="mt-2">You can customize each part independently.</p>
    </CollapsibleContent>
  </CollapsibleRoot>
)
```
