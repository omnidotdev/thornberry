import { Button } from "@/registry/thornberry/components/button";
import { Dialog, DialogCloseTrigger } from "@/registry/thornberry/components/dialog";

<Dialog
  trigger={<Button>Open Dialog</Button>}
  title="Dialog Title"
  description="This is a description of the dialog's purpose."
  className="mb-8"
>
  <div className="py-4">This is the main content of the dialog.</div>
  <div className="flex justify-end gap-2">
    <DialogCloseTrigger asChild>
      <Button variant="outline">Cancel</Button>
    </DialogCloseTrigger>
    <Button>Save</Button>
  </div>
</Dialog>

### Examples

Let's see how to use the `Dialog` component. Let's start with a basic example:

```tsx
import { Button } from "@/components/ui/button"
import { Dialog, DialogCloseTrigger } from "@/components/ui/dialog"

export const BasicDialog = () => (
  <Dialog
    trigger={<Button>Open Dialog</Button>}
    title="Dialog Title"
    description="This is a description of the dialog's purpose."
  >
    <div className="py-4">This is the main content of the dialog.</div>
    <div className="flex justify-end gap-2">
      <DialogCloseTrigger asChild>
        <Button variant="outline">Cancel</Button>
      </DialogCloseTrigger>
      <Button>Save</Button>
    </div>
  </Dialog>
)
```

For more granular control, Thornberry provides Dialog primitives:

```tsx
import { Button } from "@/components/ui/button"
import {
  DialogRoot,
  DialogTrigger,
  DialogBackdrop,
  DialogPositioner,
  DialogContent,
  DialogCloseTrigger,
  DialogTitle,
  DialogDescription
} from "@/components/ui/dialog"

export const DialogPrimitives = () => (
  <DialogRoot>
    <DialogTrigger asChild>
      <Button variant="outline">Using Primitives</Button>
    </DialogTrigger>
    <DialogBackdrop />
    <DialogPositioner>
      <DialogContent>
        <DialogCloseTrigger />
        <DialogTitle>Custom Dialog</DialogTitle>
        <DialogDescription>
          This dialog is built using primitive components directly.
        </DialogDescription>
        <div className="border-t pt-4">
          <div>You can fully customize each part of the dialog using the primitive components.</div>
        </div>
      </DialogContent>
    </DialogPositioner>
  </DialogRoot>
)
```
