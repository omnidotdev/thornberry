title: Segment Group
description: A segmented control component for selecting a single option from a group.
---

import { SegmentGroupRoot, SegmentGroupIndicator, SegmentGroupItem, SegmentGroupItemText, SegmentGroupItemControl, SegmentGroupItemHiddenInput, SegmentGroupLabel } from "@/registry/thornberry/components/segment-group";

<SegmentGroupRoot defaultValue="react">
  <SegmentGroupIndicator />
  {["React", "Vue", "Svelte"].map((framework) => (
    <SegmentGroupItem key={framework} value={framework.toLowerCase()}>
      <SegmentGroupItemText>{framework}</SegmentGroupItemText>
      <SegmentGroupItemControl />
      <SegmentGroupItemHiddenInput />
    </SegmentGroupItem>
  ))}
</SegmentGroupRoot>

## Usage

```tsx
import {
  SegmentGroupRoot,
  SegmentGroupIndicator,
  SegmentGroupItem,
  SegmentGroupItemText,
  SegmentGroupItemControl,
  SegmentGroupItemHiddenInput
} from "@/components/ui/segment-group";
```

```tsx
<SegmentGroupRoot defaultValue="react">
  <SegmentGroupIndicator />
  {["React", "Vue", "Svelte"].map((framework) => (
    <SegmentGroupItem key={framework} value={framework.toLowerCase()}>
      <SegmentGroupItemText>{framework}</SegmentGroupItemText>
      <SegmentGroupItemControl />
      <SegmentGroupItemHiddenInput />
    </SegmentGroupItem>
  ))}
</SegmentGroupRoot>
```
