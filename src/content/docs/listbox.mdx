## Usage

The listbox component requires client-side state. Import from `@/components/listbox` which includes the `"use client"` directive:

```tsx
import {
  ListboxRoot,
  ListboxLabel,
  ListboxContent,
  ListboxItem,
  ListboxItemText,
  ListboxItemIndicator,
  createListCollection,
} from "@/components/listbox";
```

```tsx
const collection = createListCollection({ 
  items: ['Option 1', 'Option 2', 'Option 3'] 
});

<ListboxRoot collection={collection}>
  <ListboxLabel>Select an option</ListboxLabel>
  <ListboxContent>
    {collection.items.map((item) => (
      <ListboxItem key={item} item={item}>
        <ListboxItemText>{item}</ListboxItemText>
        <ListboxItemIndicator />
      </ListboxItem>
    ))}
  </ListboxContent>
</ListboxRoot>
```

      ListboxRoot,

                </ListboxItem>
              ))}
            </ListboxItemGroup>
            <ListboxItemGroup>
              <ListboxItemGroupLabel>Backend</ListboxItemGroupLabel>

            </ListboxItemGroup>
