import {
  AngleSliderRoot,
  AngleSliderControl,
  AngleSliderThumb,
  AngleSliderMarkerGroup,
  AngleSliderMarker,
  AngleSliderValueText,
} from "@/registry/thornberry/components/angle-slider"


 <AngleSliderRoot defaultValue={90} className="mb-4">
    <AngleSliderControl>
      <AngleSliderThumb />
      <AngleSliderMarkerGroup>
        {[0, 45, 90, 135, 180, 225, 270, 315].map((value) => (
          <AngleSliderMarker key={value} value={value} />
        ))}
      </AngleSliderMarkerGroup>
    </AngleSliderControl>
    <AngleSliderValueText />
  </AngleSliderRoot>
