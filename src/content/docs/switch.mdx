### Examples

Let's see how to use the `Switch` components. Let's start with a basic example:

```tsx
import { 
  SwitchRoot, 
  SwitchHiddenInput, 
  SwitchControl, 
  SwitchThumb 
} from "@/components/ui/switch"

export const BasicSwitch = () => (
  <SwitchRoot>
    <SwitchHiddenInput />
    <SwitchControl>
      <SwitchThumb />
    </SwitchControl>
  </SwitchRoot>
)
```

You can add a label to your switch:

```tsx
import { 
  SwitchRoot, 
  SwitchHiddenInput, 
  SwitchControl, 
  SwitchThumb,
  SwitchLabel
} from "@/components/ui/switch"

export const LabeledSwitch = () => (
  <SwitchRoot>
    <SwitchHiddenInput />
    <SwitchControl>
      <SwitchThumb />
    </SwitchControl>
    <SwitchLabel>Airplane Mode</SwitchLabel>
  </SwitchRoot>
)
```

```tsx
import {
  SwitchRoot,
  SwitchHiddenInput,
  SwitchControl,
  SwitchThumb,
  SwitchLabel
} from "@/components/ui/switch"

export const DefaultCheckedSwitch = () => (
  <SwitchRoot defaultChecked>
    <SwitchHiddenInput />
    <SwitchControl>
      <SwitchThumb />
    </SwitchControl>
    <SwitchLabel>Custom Switch</SwitchLabel>
  </SwitchRoot>
)
```

You can customize the appearance of your switch:

```tsx
import {
  SwitchRoot,
  SwitchHiddenInput,
  SwitchControl,
  SwitchThumb,
  SwitchLabel
} from "@/components/ui/switch"

export const CustomSwitches = () => (
  <div className="flex flex-col gap-2">
    <SwitchRoot defaultChecked>
      <SwitchHiddenInput />
      <SwitchControl className="bg-red-100 data-[state=checked]:bg-red-500">
        <SwitchThumb className="bg-white" />
      </SwitchControl>
      <SwitchLabel>Custom Colors</SwitchLabel>
    </SwitchRoot>
    
    <SwitchRoot>
      <SwitchHiddenInput />
      <SwitchControl className="h-6 w-12">
        <SwitchThumb className="h-5 w-5 data-[state=checked]:translate-x-6" />
      </SwitchControl>
      <SwitchLabel>Custom Size</SwitchLabel>
    </SwitchRoot>
  </div>
)
```
