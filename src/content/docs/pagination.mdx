title: Pagination

import {
  Pagination,
  PaginationContext,
  PaginationEllipsis,
  PaginationItem,
  PaginationNextTrigger,
  PaginationPrevTrigger,
  PaginationProvider,
  PaginationRoot,
} from "@/registry/thornberry/components/pagination";

## Basic

```tsx
<Pagination count={10} />
```

<Pagination count={10} className="mb-4" />

## Current Page

```tsx
<Pagination count={10} page={3} />
```

<Pagination count={10} page={3} className="mb-4" />

## Custom Sibling Count

```tsx
<Pagination count={20} page={10} siblingCount={2} />
```

<Pagination count={20} page={10} siblingCount={2} className="mb-4" />

## Primitives

```tsx
<PaginationRoot count={10} page={4}>
  <div className="flex items-center gap-2">
    <PaginationPrevTrigger />
    {Array.from({ length: 10 }, (_, i) => i + 1).map(pageNum => (
      <PaginationItem
        key={pageNum}
        value={pageNum}
      >
        {pageNum}
      </PaginationItem>
    ))}
    <PaginationNextTrigger />
  </div>
</PaginationRoot>
```
