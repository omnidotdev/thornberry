import {
  Pagination,
  PaginationContext,
  PaginationEllipsis,
  PaginationItem,
  PaginationNextTrigger,
  PaginationPrevTrigger,
  PaginationProvider,
  PaginationRoot,
} from "@/registry/thornberry/components/pagination";

## Basic

<Pagination count={10} />

## Current Page

<Pagination count={10} page={3} />

## Large Number of Pages

<Pagination count={20} />

## Custom Sibling Count

Shows more page numbers around the current page.

<Pagination count={20} page={10} siblingCount={2} />

## Disabled

<Pagination count={10} disabled />

## Compact

<Pagination count={5} className="justify-center" />

## Custom Styling

<PaginationRoot
  count={10}
  page={5}
  className="justify-center"
>
  <div className="flex items-center gap-2">
    <PaginationPrevTrigger />
    {Array.from({ length: 10 }, (_, i) => i + 1).map(pageNum => (
      <PaginationItem 
        key={pageNum}
        value={pageNum}
        type="page"

        className="bg-amber-50 border-amber-200 data-[selected]:bg-amber-500 data-[selected]:border-amber-500 data-[selected]:text-white"
      >
        {pageNum}
      </PaginationItem>
    ))}
    <PaginationNextTrigger />
  </div>
</PaginationRoot>
