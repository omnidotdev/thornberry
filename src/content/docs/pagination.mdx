### Examples

Let's see how to use the `Pagination` component. Let's start with a basic example:

```tsx
import { Pagination } from "@/components/ui/pagination"

export const BasicPagination = () => (
  <Pagination count={10} />
)
```

You can specify the current page:

```tsx
import { Pagination } from "@/components/ui/pagination"

export const PaginationWithCurrentPage = () => (
  <Pagination count={10} page={3} />
)
```

You can customize the number of siblings displayed around the current page:

```tsx
import { Pagination } from "@/components/ui/pagination"

export const PaginationWithSiblings = () => (
  <Pagination count={20} page={10} siblingCount={2} />
)
```

For more granular control, Thornberry provides Pagination primitives:

```tsx
import {
  PaginationRoot,
  PaginationPrevTrigger,
  PaginationItem,
  PaginationNextTrigger
} from "@/components/ui/pagination"

export const PaginationPrimitives = () => (
  <PaginationRoot count={10} page={4}>
    <div className="flex items-center gap-2">
      <PaginationPrevTrigger />
      {Array.from({ length: 10 }, (_, i) => i + 1).map(pageNum => (
        <PaginationItem
          key={pageNum}
          value={pageNum}
        >
          {pageNum}
        </PaginationItem>
      ))}
      <PaginationNextTrigger />
    </div>
  </PaginationRoot>
)
```
