### Examples

Let's see how to use the `Pagination` components. Let's start with a basic example:

```tsx
import {
  PaginationRoot,
  PaginationItem,
  PaginationEllipsis,
  PaginationPrevTrigger,
  PaginationNextTrigger
} from "@/components/ui/pagination"

export const BasicPagination = () => (
  <PaginationRoot count={10}>
    <div className="flex items-center gap-1">
      <PaginationPrevTrigger />
      <PaginationItem value={1}>1</PaginationItem>
      <PaginationItem value={2}>2</PaginationItem>
      <PaginationItem value={3}>3</PaginationItem>
      <PaginationEllipsis index={0} />
      <PaginationItem value={10}>10</PaginationItem>
      <PaginationNextTrigger />
    </div>
  </PaginationRoot>
)
```

You can specify the current page:

```tsx
import {
  PaginationRoot,
  PaginationItem,
  PaginationEllipsis,
  PaginationPrevTrigger,
  PaginationNextTrigger
} from "@/components/ui/pagination"

export const PaginationWithCurrentPage = () => (
  <PaginationRoot count={10} page={3}>
    <div className="flex items-center gap-1">
      <PaginationPrevTrigger />
      <PaginationItem value={1}>1</PaginationItem>
      <PaginationItem value={2}>2</PaginationItem>
      <PaginationItem value={3}>3</PaginationItem>
      <PaginationItem value={4}>4</PaginationItem>
      <PaginationItem value={5}>5</PaginationItem>
      <PaginationEllipsis index={0} />
      <PaginationItem value={10}>10</PaginationItem>
      <PaginationNextTrigger />
    </div>
  </PaginationRoot>
)
```

You can customize the number of items displayed:

```tsx
import {
  PaginationRoot,
  PaginationPrevTrigger,
  PaginationItem,
  PaginationNextTrigger
} from "@/components/ui/pagination"

export const CustomPagination = () => (
  <PaginationRoot count={10} page={4}>
    <div className="flex items-center gap-2">
      <PaginationPrevTrigger />
      {Array.from({ length: 10 }, (_, i) => i + 1).map(pageNum => (
        <PaginationItem
          key={pageNum}
          value={pageNum}
        >
          {pageNum}
        </PaginationItem>
      ))}
      <PaginationNextTrigger />
    </div>
  </PaginationRoot>
)
```
