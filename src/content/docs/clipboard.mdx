## Usage

```tsx
import {
  ClipboardRoot,
  ClipboardLabel,
  ClipboardControl,
  ClipboardInput,
  ClipboardTrigger
} from "@/components/ui/clipboard";
```

```tsx
<ClipboardRoot value="Text to copy">
  <ClipboardLabel>Label</ClipboardLabel>
  <ClipboardControl>
    <ClipboardInput />
    <ClipboardTrigger />
  </ClipboardControl>
</ClipboardRoot>
```
