import { DownloadTrigger } from "@/registry/thornberry/components/download-trigger";

<DownloadTrigger 
  data="Hello, World!" 
  fileName="example.txt"

  className="mb-8"
/>

## Usage

```tsx
import { DownloadTrigger } from "@/components/ui/download-trigger";
```

```tsx
<DownloadTrigger 
  data="File content" 
  fileName="file.txt" 
  mimeType="text/plain"
/>
```
