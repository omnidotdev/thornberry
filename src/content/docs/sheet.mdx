import { 
  SheetRoot,
  SheetTrigger,
  SheetBackdrop,
  SheetPositioner,
  SheetContent,
  SheetTitle,
  SheetDescription,
  SheetCloseTrigger,
} from "@/registry/thornberry/components/sheet";

import { X } from "lucide-react";

      <SheetCloseTrigger />

        <Button>Save</Button>
      </div>
    </SheetContent>
  </SheetPositioner>
</SheetRoot>
