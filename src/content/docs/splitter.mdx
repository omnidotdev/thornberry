---
title: Splitter
description: Resizable split panels with adjustable layout, supporting both horizontal and vertical orientations.
---

import { SplitterRoot, SplitterPanel, SplitterResizeTrigger } from "@/registry/thornberry/components/splitter";

<SplitterRoot panels={[{ id: "a" }, { id: "b" }]} className="h-64 w-full rounded-md border">
  <SplitterPanel id="a" className="flex items-center justify-center bg-muted/50">
    <p className="text-muted-foreground text-sm">Panel A</p>
  </SplitterPanel>
  <SplitterResizeTrigger id="a:b" />
  <SplitterPanel id="b" className="flex items-center justify-center bg-muted/30">
    <p className="text-muted-foreground text-sm">Panel B</p>
  </SplitterPanel>
</SplitterRoot>

## Usage

```tsx
import {
  SplitterRoot,
  SplitterPanel,
  SplitterResizeTrigger
} from "@/components/ui/splitter";
```

```tsx
<SplitterRoot panels={[{ id: "a" }, { id: "b" }]}>
  <SplitterPanel id="a">
    Panel A content
  </SplitterPanel>
  <SplitterResizeTrigger id="a:b" />
  <SplitterPanel id="b">
    Panel B content
  </SplitterPanel>
</SplitterRoot>
```
