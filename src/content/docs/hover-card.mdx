---
title: Hover Card
description: A card that appears when hovering over a trigger element, providing additional information.
---

import { FiUser, FiMail, FiMapPin } from "react-icons/fi";
import { Badge } from "@/registry/thornberry/components/badge";
import { HoverCardRoot, HoverCardTrigger, HoverCardContent, HoverCardPositioner, HoverCardArrow, HoverCardArrowTip } from "@/registry/thornberry/components/hover-card";

<HoverCardRoot>
  <HoverCardTrigger className="mb-8">
    <Badge variant="outline">Hover me</Badge>
  </HoverCardTrigger>
  <HoverCardPositioner>
    <HoverCardContent>
      <HoverCardArrow>
        <HoverCardArrowTip />
      </HoverCardArrow>
      <p>Here is the hover card content</p>
    </HoverCardContent>
  </HoverCardPositioner>
</HoverCardRoot>

## Usage

```tsx
import {
  HoverCardRoot,
  HoverCardTrigger,
  HoverCardContent,
  HoverCardPositioner,
  HoverCardArrow,
  HoverCardArrowTip
} from "@/components/ui/hover-card";
```

```tsx
<HoverCardRoot>
  <HoverCardTrigger>
    Hover me
  </HoverCardTrigger>
  <HoverCardPositioner>
    <HoverCardContent>
      <HoverCardArrow>
        <HoverCardArrowTip />
      </HoverCardArrow>
      Hover card content
    </HoverCardContent>
  </HoverCardPositioner>
</HoverCardRoot>
```
