---
title: Hover Card
description: A card that appears when hovering over a trigger element, providing additional information.
---

import { Badge } from "@/registry/thornberry/components/badge";
import { HoverCardRoot, HoverCardTrigger, HoverCardContent, HoverCardPositioner, HoverCardArrow, HoverCardArrowTip } from "@/registry/thornberry/components/hover-card";

<HoverCardRoot>
  <HoverCardTrigger>
    <Badge variant="outline" className="mb-8">Hover me</Badge>
  </HoverCardTrigger>
  
  <HoverCardPositioner>
    <HoverCardContent>
      <HoverCardArrow>
        <HoverCardArrowTip />
      </HoverCardArrow>
      
      Here is the content
    </HoverCardContent>
  </HoverCardPositioner>
</HoverCardRoot>

### Examples

Let's see how to use the `HoverCard` components. Here's a basic example:

```tsx
import { 
  HoverCardRoot, 
  HoverCardTrigger, 
  HoverCardContent, 
  HoverCardPositioner,
  HoverCardArrow,
  HoverCardArrowTip
} from "@/components/ui/hover-card"

export const BasicHoverCard = () => (
  <HoverCardRoot>
    <HoverCardTrigger>
      Hover me
    </HoverCardTrigger>
    
    <HoverCardPositioner>
      <HoverCardContent>
        <HoverCardArrow>
          <HoverCardArrowTip />
        </HoverCardArrow>
        
        Hover Card
        Additional information here.
      </HoverCardContent>
    </HoverCardPositioner>
  </HoverCardRoot>
)
```

You can customize the delay before showing or hiding the hover card:

```tsx
import { 
  HoverCardRoot, 
  HoverCardTrigger, 
  HoverCardContent, 
  HoverCardPositioner,
  HoverCardArrow,
  HoverCardArrowTip
} from "@/components/ui/hover-card"

export const DelayedHoverCard = () => (
  <HoverCardRoot openDelay={300} closeDelay={200}>
    <HoverCardTrigger>
      Hover with delay
    </HoverCardTrigger>
    
    <HoverCardPositioner>
      <HoverCardContent>
        <HoverCardArrow>
          <HoverCardArrowTip />
        </HoverCardArrow>
        
        Delayed Hover Card
        This hover card has custom delay timing.
      </HoverCardContent>
    </HoverCardPositioner>
  </HoverCardRoot>
)
```

You can hide the arrow by not including the HoverCardArrow component:

```tsx
import { 
  HoverCardRoot, 
  HoverCardTrigger, 
  HoverCardContent, 
  HoverCardPositioner
} from "@/components/ui/hover-card"

export const HoverCardWithoutArrow = () => (
  <HoverCardRoot>
    <HoverCardTrigger>
      No arrow
    </HoverCardTrigger>
    
    <HoverCardPositioner>
      <HoverCardContent>
        No Arrow
        This hover card has no arrow pointer.
      </HoverCardContent>
    </HoverCardPositioner>
  </HoverCardRoot>
)
```
