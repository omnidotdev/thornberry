## Usage

```tsx
import {
  ComboboxRoot,
  ComboboxControl,
  ComboboxInput,
  ComboboxTrigger,
  ComboboxPositioner,
  ComboboxContent,
  ComboboxItem
} from "@/components/ui/combobox";
```

```tsx
<ComboboxRoot items={[{ label: "Item 1", value: "item-1" }]}>
  <ComboboxControl>
    <ComboboxInput placeholder="Select an item..." />
    <ComboboxTrigger />
  </ComboboxControl>
  <ComboboxPositioner>
    <ComboboxContent>
      <ComboboxItem item={{ label: "Item 1", value: "item-1" }}>
        Item 1
      </ComboboxItem>
    </ComboboxContent>
  </ComboboxPositioner>
</ComboboxRoot>
```
