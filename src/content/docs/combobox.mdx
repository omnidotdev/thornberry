## Basic

```tsx
<Combobox
  items={[
    { label: "Apple", value: "apple" },
    { label: "Banana", value: "banana" },
    { label: "Orange", value: "orange" },
    { label: "Strawberry", value: "strawberry" },
    { label: "Blueberry", value: "blueberry" }
  ]}
/>
```

## Primitives

```tsx
<ComboboxRoot
  items={[
    { label: "Red", value: "red" },
    { label: "Green", value: "green" },
    { label: "Blue", value: "blue" },
    { label: "Yellow", value: "yellow" },
    { label: "Purple", value: "purple" }
  ]}
>
  <ComboboxControl>
    <ComboboxInput placeholder="Search colors..." />
    <ComboboxTrigger />
  </ComboboxControl>
  <ComboboxPositioner>
    <ComboboxContent>
      <ComboboxItem item={{ label: "Red", value: "red" }}>Red</ComboboxItem>
      <ComboboxItem item={{ label: "Green", value: "green" }}>Green</ComboboxItem>
      <ComboboxItem item={{ label: "Blue", value: "blue" }}>Blue</ComboboxItem>
      <ComboboxItem item={{ label: "Yellow", value: "yellow" }}>Yellow</ComboboxItem>
      <ComboboxItem item={{ label: "Purple", value: "purple" }}>Purple</ComboboxItem>
    </ComboboxContent>
  </ComboboxPositioner>
</ComboboxRoot>
```
