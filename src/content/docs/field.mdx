## Usage

```tsx
import {
  FieldRoot,
  FieldLabel,
  FieldInput,
  FieldHelperText,
  FieldErrorText
} from "@/components/ui/field";
```

```tsx
<FieldRoot>
  <FieldLabel>Label</FieldLabel>
  <FieldInput />
  <FieldHelperText>Helper text</FieldHelperText>
  <FieldErrorText>Error message</FieldErrorText>
</FieldRoot>
```
