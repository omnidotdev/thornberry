import { Tooltip } from "@/registry/thornberry/components/tooltip";
import { Button } from "@/registry/thornberry/components/button";

<Tooltip content="This is a tooltip" className="mb-8">
  <Button variant="outline">Hover Me</Button>
</Tooltip>

### Examples

Let's see how to use the `Tooltip` component. Let's start with a basic example:

```tsx
import { Tooltip } from "@/components/ui/tooltip"
import { Button } from "@/components/ui/button"

export const BasicTooltip = () => (
  <Tooltip content="This is a tooltip">
    <Button variant="outline">Hover Me</Button>
  </Tooltip>
)
```

You can customize the placement of the tooltip:

```tsx
import { Tooltip } from "@/components/ui/tooltip"
import { Button } from "@/components/ui/button"

export const TooltipPlacements = () => (
  <div className="flex gap-4">
    <Tooltip content="Left tooltip" positioning={{ placement: "left" }}>
      <Button variant="outline" size="sm">Left</Button>
    </Tooltip>
    
    <Tooltip content="Top tooltip" positioning={{ placement: "top" }}>
      <Button variant="outline" size="sm">Top</Button>
    </Tooltip>
    
    <Tooltip content="Bottom tooltip" positioning={{ placement: "bottom" }}>
      <Button variant="outline" size="sm">Bottom</Button>
    </Tooltip>
    
    <Tooltip content="Right tooltip" positioning={{ placement: "right" }}>
      <Button variant="outline" size="sm">Right</Button>
    </Tooltip>
  </div>
)
```

You can configure open and close delay times:

```tsx
import { Tooltip } from "@/components/ui/tooltip"
import { Button } from "@/components/ui/button"

export const TooltipWithDelays = () => (
  <div className="flex gap-4">
    <Tooltip
      content="Fast open (100ms), slow close (1000ms)"
      openDelay={100}
      closeDelay={1000}
    >
      <Button variant="outline" size="sm">
        Slow Close
      </Button>
    </Tooltip>

    <Tooltip
      content="Slow open (800ms), fast close (100ms)"
      openDelay={800}
      closeDelay={100}
    >
      <Button variant="outline" size="sm">
        Slow Open
      </Button>
    </Tooltip>
  </div>
)
```

For more granular control, Thornberry provides Tooltip primitives:

```tsx
import {
  TooltipRoot,
  TooltipTrigger,
  TooltipPositioner,
  TooltipContent,
  TooltipArrow,
  TooltipArrowTip
} from "@/components/ui/tooltip"
import { Button } from "@/components/ui/button"

export const TooltipPrimitives = () => (
  <TooltipRoot>
    <TooltipTrigger asChild>
      <Button variant="outline">Hover Me</Button>
    </TooltipTrigger>
    <TooltipPositioner>
      <TooltipContent>
        <TooltipArrow>
          <TooltipArrowTip />
        </TooltipArrow>
        <div className="flex items-center gap-2">
          <div className="h-2 w-2 rounded-full bg-blue-500" />
          <div>Custom styled tooltip using primitives</div>
        </div>
      </TooltipContent>
    </TooltipPositioner>
  </TooltipRoot>
)
```
