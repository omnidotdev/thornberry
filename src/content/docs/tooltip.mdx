---
title: Tooltip
description: A popup that displays information related to an element when it receives focus or is hovered over.
---

import { FiInfo, FiAlertCircle, FiHelpCircle, FiSettings } from "react-icons/fi";
import { TooltipRoot, TooltipTrigger, TooltipPositioner, TooltipContent, TooltipArrow, TooltipArrowTip } from "@/registry/thornberry/components/tooltip";
import { Button } from "@/registry/thornberry/components/button";

<TooltipRoot>
  <TooltipTrigger asChild className="mb-8">
    <Button variant="outline">Hover Me</Button>
  </TooltipTrigger>
  <TooltipPositioner>
    <TooltipContent>
      This is a tooltip
    </TooltipContent>
  </TooltipPositioner>
</TooltipRoot>

## Installation

import { InstallationTabs } from "@/components";

<InstallationTabs filename="tooltip.json" />

## Usage

```tsx
import {
  TooltipRoot,
  TooltipTrigger,
  TooltipPositioner,
  TooltipContent
} from "@/components/ui/tooltip";
import { Button } from "@/components/ui/button";
```

```tsx
<TooltipRoot>
  <TooltipTrigger asChild>
    <Button>Hover Me</Button>
  </TooltipTrigger>
  <TooltipPositioner>
    <TooltipContent>
      Tooltip content
    </TooltipContent>
  </TooltipPositioner>
</TooltipRoot>
```
