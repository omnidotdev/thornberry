## Usage

```tsx
import {
  TooltipRoot,
  TooltipTrigger,
  TooltipPositioner,
  TooltipContent
} from "@/components/ui/tooltip";
import { Button } from "@/components/ui/button";
```

```tsx
<TooltipRoot>
  <TooltipTrigger asChild>
    <Button>Hover Me</Button>
  </TooltipTrigger>
  <TooltipPositioner>
    <TooltipContent>
      Tooltip content
    </TooltipContent>
  </TooltipPositioner>
</TooltipRoot>
```
