### Examples

Let's see how to use the `Tooltip` components. Let's start with a basic example:

```tsx
import {
  TooltipRoot,
  TooltipTrigger,
  TooltipPositioner,
  TooltipContent
} from "@/components/ui/tooltip"
import { Button } from "@/components/ui/button"

export const BasicTooltip = () => (
  <TooltipRoot>
    <TooltipTrigger asChild>
      <Button variant="outline">Hover Me</Button>
    </TooltipTrigger>
    <TooltipPositioner>
      <TooltipContent>
        This is a tooltip
      </TooltipContent>
    </TooltipPositioner>
  </TooltipRoot>
)
```

You can customize the placement of the tooltip:

```tsx
import {
  TooltipRoot,
  TooltipTrigger,
  TooltipPositioner,
  TooltipContent
} from "@/components/ui/tooltip"
import { Button } from "@/components/ui/button"

export const TooltipPlacements = () => (
  <div className="flex gap-4">
    <TooltipRoot>
      <TooltipTrigger asChild>
        <Button variant="outline" size="sm">Left</Button>
      </TooltipTrigger>
      <TooltipPositioner placement="left">
        <TooltipContent>
          Left tooltip
        </TooltipContent>
      </TooltipPositioner>
    </TooltipRoot>

    <TooltipRoot>
      <TooltipTrigger asChild>
        <Button variant="outline" size="sm">Top</Button>
      </TooltipTrigger>
      <TooltipPositioner placement="top">
        <TooltipContent>
          Top tooltip
        </TooltipContent>
      </TooltipPositioner>
    </TooltipRoot>

    <TooltipRoot>
      <TooltipTrigger asChild>
        <Button variant="outline" size="sm">Bottom</Button>
      </TooltipTrigger>
      <TooltipPositioner placement="bottom">
        <TooltipContent>
          Bottom tooltip
        </TooltipContent>
      </TooltipPositioner>
    </TooltipRoot>

    <TooltipRoot>
      <TooltipTrigger asChild>
        <Button variant="outline" size="sm">Right</Button>
      </TooltipTrigger>
      <TooltipPositioner placement="right">
        <TooltipContent>
          Right tooltip
        </TooltipContent>
      </TooltipPositioner>
    </TooltipRoot>
  </div>
)
```

You can configure open and close delay times:

```tsx
import {
  TooltipRoot,
  TooltipTrigger,
  TooltipPositioner,
  TooltipContent
} from "@/components/ui/tooltip"
import { Button } from "@/components/ui/button"

export const TooltipWithDelays = () => (
  <div className="flex gap-4">
    <TooltipRoot openDelay={100} closeDelay={1000}>
      <TooltipTrigger asChild>
        <Button variant="outline" size="sm">
          Slow Close
        </Button>
      </TooltipTrigger>
      <TooltipPositioner>
        <TooltipContent>
          Fast open (100ms), slow close (1000ms)
        </TooltipContent>
      </TooltipPositioner>
    </TooltipRoot>

    <TooltipRoot openDelay={800} closeDelay={100}>
      <TooltipTrigger asChild>
        <Button variant="outline" size="sm">
          Slow Open
        </Button>
      </TooltipTrigger>
      <TooltipPositioner>
        <TooltipContent>
          Slow open (800ms), fast close (100ms)
        </TooltipContent>
      </TooltipPositioner>
    </TooltipRoot>
  </div>
)
```
