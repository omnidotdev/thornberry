import { FiUser } from "react-icons/fi";

import { AvatarRoot, AvatarFallback, AvatarImage } from "@/registry/thornberry/components/avatar";

<AvatarRoot className="mb-8">
  <AvatarImage src="/img/elon-musk.jpg" alt="Elon Musk" />
  <AvatarFallback>EM</AvatarFallback>
</AvatarRoot>

## Installation

<Tabs items={["bun", "pnpm", "yarn", "npm"]} className="mt-4">
  <Tab value="bun">
    ```
    bunx --bun shadcn@latest add http://localhost:3000/r/avatar.json
    ```
  </Tab>
  <Tab value="pnpm">
    ```
    pnpm dlx shadcn@latest add http://localhost:3000/r/avatar.json
    ```
  </Tab>
  <Tab value="yarn">
    ```
    npx shadcn@latest add http://localhost:3000/r/avatar.json
    ```
  </Tab>
  <Tab value="npm">
    ```
    npx shadcn@latest add http://localhost:3000/r/avatar.json
    ```
  </Tab>
</Tabs>

## Usage

```tsx
import { AvatarRoot, AvatarFallback, AvatarImage } from "@/components/ui/avatar";
```

```tsx
<AvatarRoot>
  <AvatarImage src="/path-to-image.jpg" alt="User name" />
  <AvatarFallback>UN</AvatarFallback>
</AvatarRoot>
```
