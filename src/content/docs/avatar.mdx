### Examples

Let's see how to use the `Avatar` components. Let's start with a basic example:

```tsx
import { AvatarRoot, AvatarFallback, AvatarImage } from "@/components/ui/avatar"

export const BasicAvatar = () => (
  <AvatarRoot>
    <AvatarFallback>EM</AvatarFallback>
    <AvatarImage src="/img/elon-musk.jpg" alt="Elon Musk" />
  </AvatarRoot>
)
```

You can provide a custom fallback when the image fails to load:

```tsx
import { AvatarRoot, AvatarFallback, AvatarImage } from "@/components/ui/avatar"
import { UserIcon } from "lucide-react"

export const FallbackAvatar = () => (
  <AvatarRoot>
    <AvatarFallback>
      <UserIcon className="h-6 w-6" />
    </AvatarFallback>
    <AvatarImage src="/img/elon-musk.jpg" alt="User" />
  </AvatarRoot>
)
```

```tsx
import { AvatarRoot, AvatarFallback, AvatarImage } from "@/components/ui/avatar"

export const CustomizedAvatar = () => (
  <AvatarRoot className="h-16 w-16 border-4 border-primary">
    <AvatarFallback className="bg-primary text-white text-lg">
      JD
    </AvatarFallback>
    <AvatarImage src="/img/john-doe.jpg" alt="John Doe" />
  </AvatarRoot>
)
```
