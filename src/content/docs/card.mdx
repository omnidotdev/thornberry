### Examples

Let's see how to use the `Card` component. Let's start with a basic example:

```tsx
import { Card } from "@/components/ui/card"
import { Button } from "@/components/ui/button"

export const BasicCard = () => (
  <Card
    header={{
      title: "Card Title",
      description: "This is the card description."
    }}
    footer={<Button>Save</Button>}
  >
    This card uses the simplified API with header and footer props.
  </Card>
)
```

For more granular control, Thornberry provides Card primitives:

```tsx
import {
  CardRoot,
  CardHeader,
  CardTitle,
  CardDescription,
  CardContent,
  CardFooter
} from "@/components/ui/card"
import { Button } from "@/components/ui/button"

export const CardWithPrimitives = () => (
  <CardRoot>
    <CardHeader>
      <CardTitle>Card Title</CardTitle>
      <CardDescription>Card Description</CardDescription>
    </CardHeader>
    <CardContent>
      This is the content area of the card, where the main information is displayed.
    </CardContent>
    <CardFooter>
      <Button>Action</Button>
    </CardFooter>
  </CardRoot>
)
```
