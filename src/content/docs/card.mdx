### Examples

Let's see how to use the `Card` components. Let's start with a basic example:

```tsx
import {
  CardRoot,
  CardHeader,
  CardTitle,
  CardDescription,
  CardContent,
  CardFooter
} from "@/components/ui/card"
import { Button } from "@/components/ui/button"

export const BasicCard = () => (
  <CardRoot>
    <CardHeader>
      <CardTitle>Card Title</CardTitle>

      <CardDescription>This is the card description.</CardDescription>
    </CardHeader>

    <CardContent>
      This is the content area of the card, where the main information is displayed.
    </CardContent>

    <CardFooter>
      <Button>Save</Button>
    </CardFooter>
  </CardRoot>
)
```

```tsx
import {
  CardRoot,

```
