## Installation

<Tabs items={["bun", "pnpm", "yarn", "npm"]} className="mt-4">
  <Tab value="bun">
    ```
    bunx --bun shadcn@latest add http://localhost:3000/r/badge.json
    ```
  </Tab>
  <Tab value="pnpm">
    ```
    pnpm dlx shadcn@latest add http://localhost:3000/r/badge.json
    ```
  </Tab>
  <Tab value="yarn">
    ```
    npx shadcn@latest add http://localhost:3000/r/badge.json
    ```
  </Tab>
  <Tab value="npm">
    ```
    npx shadcn@latest add http://localhost:3000/r/badge.json
    ```
  </Tab>
</Tabs>

## Usage

```tsx
import { Badge } from "@/components/ui/badge";
```

```tsx
<Badge variant="outline">Badge</Badge>
```
