import { SliderRoot, SliderControl, SliderTrack, SliderRange, SliderThumb } from "@/registry/thornberry/components/slider";

<SliderRoot defaultValue={[50]} className="mb-8">
  <SliderControl>
    <SliderTrack>
      <SliderRange />
    </SliderTrack>

    <SliderThumb index={0} />
  </SliderControl>
</SliderRoot>

### Examples

Let's see how to use the `Slider` components. Let's start with a basic example:

```tsx
import { 
  SliderRoot, 
  SliderControl, 
  SliderTrack, 
  SliderRange, 
  SliderThumb 
} from "@/components/ui/slider"

export const BasicSlider = () => (
  <SliderRoot defaultValue={[50]}>
    <SliderControl>
      <SliderTrack>
        <SliderRange />
      </SliderTrack>

      <SliderThumb index={0} />
    </SliderControl>
  </SliderRoot>
)
```

You can add a label to your slider:

```tsx
import { 
  SliderRoot, 
  SliderLabel,
  SliderControl, 
  SliderTrack, 
  SliderRange, 
  SliderThumb 
} from "@/components/ui/slider"

export const LabeledSlider = () => (
  <SliderRoot defaultValue={[75]}>
    <SliderLabel>Volume</SliderLabel>

    <SliderControl>
      <SliderTrack>
        <SliderRange />
      </SliderTrack>

      <SliderThumb index={0} />
    </SliderControl>
  </SliderRoot>
)
```

You can display the current value:

```tsx
import { 
  SliderRoot, 
  SliderLabel,
  SliderControl, 
  SliderTrack, 
  SliderRange, 
  SliderThumb,
  SliderValueText
} from "@/components/ui/slider"

export const SliderWithValue = () => (
  <SliderRoot defaultValue={[60]}>
    <SliderLabel>Brightness</SliderLabel>

    <SliderControl>
      <SliderTrack>
        <SliderRange />
      </SliderTrack>

      <SliderThumb index={0} />
    </SliderControl>

    <SliderValueText />
  </SliderRoot>
)
```

You can customize the range with min, max, and step values:

```tsx
import { 
  SliderRoot, 
  SliderLabel,
  SliderControl, 
  SliderTrack, 
  SliderRange, 
  SliderThumb,
  SliderValueText
} from "@/components/ui/slider"

export const CustomRangeSlider = () => (
  <SliderRoot
    min={15}
    max={30}
    step={0.5}
    defaultValue={[22.5]}
  >
    <SliderLabel>Temperature</SliderLabel>

    <SliderControl>
      <SliderTrack>
        <SliderRange />
      </SliderTrack>

      <SliderThumb index={0} />
    </SliderControl>

    <SliderValueText />
  </SliderRoot>
)
```

You can add markers to your slider:

```tsx
import { 
  SliderRoot, 
  SliderLabel,
  SliderControl, 
  SliderTrack, 
  SliderRange, 
  SliderThumb,
  SliderValueText,
  SliderMarker
} from "@/components/ui/slider"

export const SliderWithMarkers = () => (
  <SliderRoot
    min={0}
    max={100}
    step={25}
    defaultValue={[50]}
  >
    <SliderLabel>Speed</SliderLabel>

    <SliderControl>
      <SliderTrack>
        <SliderRange />
      </SliderTrack>

      <SliderMarker value={0}>0%</SliderMarker>

      <SliderMarker value={25}>25%</SliderMarker>

      <SliderMarker value={50}>50%</SliderMarker>

      <SliderMarker value={75}>75%</SliderMarker>

      <SliderMarker value={100}>100%</SliderMarker>

      <SliderThumb index={0} />
    </SliderControl>

    <SliderValueText />
  </SliderRoot>
)
```

You can create a range slider with multiple thumbs:

```tsx
import {
  SliderRoot,
  SliderLabel,
  SliderControl,
  SliderTrack,
  SliderRange,
  SliderThumb,
  SliderValueText
} from "@/components/ui/slider"

export const RangeSlider = () => (
  <SliderRoot
    min={0}
    max={100}
    step={10}
    defaultValue={[30, 70]}
  >
    <SliderLabel>Range Slider</SliderLabel>

    <SliderControl>
      <SliderTrack>
        <SliderRange />
      </SliderTrack>

      <SliderThumb index={0} />

      <SliderThumb index={1} />
    </SliderControl>

    <SliderValueText />
  </SliderRoot>
)
```
