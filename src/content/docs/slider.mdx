---
title: Slider
description: A control that allows users to select a value or range from a given range
---

import {
  Slider,
  SliderContext,
  SliderControl,
  SliderLabel,
  SliderMarker,
  SliderProvider,
  SliderRange,
  SliderRoot,
  SliderThumb,
  SliderTrack,
  SliderValueText,
} from "@/registry/thornberry/components/slider";

## Basic

```tsx
<Slider defaultValue={[50]} />
```

<Slider defaultValue={[50]} className="mb-4" />

## With Label

```tsx
<Slider label="Volume" defaultValue={[75]} />
```

<Slider label="Volume" defaultValue={[75]} className="mb-4" />

## With Value Display

```tsx
<Slider label="Brightness" defaultValue={[60]} showValueText />
```

<Slider label="Brightness" defaultValue={[60]} showValueText className="mb-4" />

## Custom Range

```tsx
<Slider
  label="Temperature"
  min={15}
  max={30}
  step={0.5}
  defaultValue={[22.5]}
  showValueText
/>
```

<Slider
  label="Temperature"
  min={15}
  max={30}
  step={0.5}
  defaultValue={[22.5]}
  showValueText
  className="mb-4"
/>

## With Markers

```tsx
<Slider
  label="Speed"
  min={0}
  max={100}
  step={25}
  defaultValue={[50]}
  markers={[
    { value: 0, label: "0%" },
    { value: 25, label: "25%" },
    { value: 50, label: "50%" },
    { value: 75, label: "75%" },
    { value: 100, label: "100%" }
  ]}
  showValueText
/>
```

## Range Slider

```tsx
<SliderRoot
  min={0}
  max={100}
  step={10}
  defaultValue={[30, 70]}
>
  <SliderLabel>Range Slider</SliderLabel>
  <SliderControl>
    <SliderTrack>
      <SliderRange />
    </SliderTrack>
    <SliderThumb index={0} />
    <SliderThumb index={1} />
  </SliderControl>
  <SliderValueText />
</SliderRoot>
```

<SliderRoot

</SliderRoot>

  <SliderValueText />
