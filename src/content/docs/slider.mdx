---
title: Slider
description: A control that allows users to select a value or range from a given range
---

## Basic

<Slider defaultValue={[50]} />

## With Label

<Slider label="Volume" defaultValue={[75]} />

## With Value Display

<Slider label="Brightness" defaultValue={[60]} showValueText />

## Custom Range

<Slider 
  label="Temperature"
  min={15}
  max={30}
  step={0.5}
  defaultValue={[22.5]}
  showValueText
/>

## With Markers

<Slider 
  label="Speed"
  min={0}
  max={100}
  step={25}
  defaultValue={[50]}
  markers={[
    { value: 0, label: "0%" },
    { value: 25, label: "25%" },
    { value: 50, label: "50%" },
    { value: 75, label: "75%" },
    { value: 100, label: "100%" }
  ]}
  showValueText
/>

## Disabled

<Slider 
  label="Disabled Slider" 
  defaultValue={[30]} 
  disabled 
  showValueText
/>
