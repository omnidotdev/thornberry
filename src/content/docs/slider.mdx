title: Slider

import { FiVolume2, FiSun, FiMoon, FiThermometer } from "react-icons/fi";
import { SliderRoot, SliderControl, SliderTrack, SliderRange, SliderThumb, SliderLabel, SliderValueText, SliderMarker } from "@/registry/thornberry/components/slider";

    <SliderTrack>
      <SliderRange />

  </SliderControl>
</SliderRoot>

## Installation

<Tabs items={["bun", "pnpm", "yarn", "npm"]} className="mt-4">

    ```
    bunx --bun shadcn@latest add http://localhost:3000/r/slider.json
    ```
  </Tab>
  <Tab value="pnpm">
    ```
    pnpm dlx shadcn@latest add http://localhost:3000/r/slider.json
    ```
  </Tab>
  <Tab value="yarn">
    ```
    npx shadcn@latest add http://localhost:3000/r/slider.json
    ```
  </Tab>
  <Tab value="npm">
    ```
    npx shadcn@latest add http://localhost:3000/r/slider.json
    ```
  </Tab>
</Tabs>

## Usage

```tsx
import {
  SliderRoot,
  SliderControl,
  SliderTrack,
  SliderRange,
  SliderThumb
} from "@/components/ui/slider";
```

```tsx
<SliderRoot defaultValue={[50]}>
  <SliderControl>
    <SliderTrack>
      <SliderRange />
    </SliderTrack>
    <SliderThumb index={0} />
  </SliderControl>
</SliderRoot>
```
