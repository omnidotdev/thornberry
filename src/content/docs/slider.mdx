### Examples

Let's see how to use the `Slider` component. Let's start with a basic example:

```tsx
import { Slider } from "@/components/ui/slider"

export const BasicSlider = () => (
  <Slider defaultValue={[50]} />
)
```

You can add a label to your slider:

```tsx
import { Slider } from "@/components/ui/slider"

export const LabeledSlider = () => (
  <Slider label="Volume" defaultValue={[75]} />
)
```

You can display the current value:

```tsx
import { Slider } from "@/components/ui/slider"

export const SliderWithValue = () => (
  <Slider 
    label="Brightness" 
    defaultValue={[60]} 
    showValueText 
  />
)
```

You can customize the range with min, max, and step values:

```tsx
import { Slider } from "@/components/ui/slider"

export const CustomRangeSlider = () => (
  <Slider
    label="Temperature"
    min={15}
    max={30}
    step={0.5}
    defaultValue={[22.5]}
    showValueText
  />
)
```

You can add markers to your slider:

```tsx
import { Slider } from "@/components/ui/slider"

export const SliderWithMarkers = () => (
  <Slider
    label="Speed"
    min={0}
    max={100}
    step={25}
    defaultValue={[50]}
    markers={[
      { value: 0, label: "0%" },
      { value: 25, label: "25%" },
      { value: 50, label: "50%" },
      { value: 75, label: "75%" },
      { value: 100, label: "100%" }
    ]}
    showValueText
  />
)
```

For more granular control, Thornberry provides Slider primitives:

```tsx
import {
  SliderRoot,
  SliderLabel,
  SliderControl,
  SliderTrack,
  SliderRange,
  SliderThumb,
  SliderValueText,
  SliderMarker
} from "@/components/ui/slider"

export const SliderPrimitives = () => (
  <SliderRoot
    min={0}
    max={100}
    defaultValue={[40]}
  >
    <SliderLabel>Custom Slider</SliderLabel>
    <SliderControl>
      <SliderTrack>
        <SliderRange />
      </SliderTrack>
      <SliderMarker value={0}>Min</SliderMarker>
      <SliderMarker value={50}>Mid</SliderMarker>
      <SliderMarker value={100}>Max</SliderMarker>
      <SliderThumb index={0} />
    </SliderControl>
    <SliderValueText />
  </SliderRoot>
)
```

You can create a range slider with multiple thumbs:

```tsx
import {
  SliderRoot,
  SliderLabel,
  SliderControl,
  SliderTrack,
  SliderRange,
  SliderThumb,
  SliderValueText
} from "@/components/ui/slider"

export const RangeSlider = () => (
  <SliderRoot
    min={0}
    max={100}
    step={10}
    defaultValue={[30, 70]}
  >
    <SliderLabel>Range Slider</SliderLabel>
    <SliderControl>
      <SliderTrack>
        <SliderRange />
      </SliderTrack>
      <SliderThumb index={0} />
      <SliderThumb index={1} />
    </SliderControl>
    <SliderValueText />
  </SliderRoot>
)
```
