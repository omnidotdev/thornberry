title: Steps
description: Display a sequence of steps to guide users through a multi-step process.
---

import { Check } from "lucide-react";
import { StepsRoot, StepsList, StepsItem, StepsTrigger, StepsIndicator, StepsSeparator, StepsContent, StepsCompletedContent, StepsNextTrigger, StepsPrevTrigger } from "@/registry/thornberry/components/steps";

export const items = [
  { value: 'first', title: 'Step 1', description: 'Contact Info' },
  { value: 'second', title: 'Step 2', description: 'Date & Time' },
  { value: 'third', title: 'Step 3', description: 'Confirmation' },
];

<StepsRoot count={items.length} defaultStep={0}>
  <StepsList>
    {items.map((item, index) => (
      <StepsItem key={index} index={index}>
        <StepsTrigger>
          <StepsIndicator>
            {index + 1}
          </StepsIndicator>
        </StepsTrigger>
        <StepsSeparator />
      </StepsItem>
    ))}
  </StepsList>
</StepsRoot>

## Usage

```tsx
import {
  StepsRoot,
  StepsList,
  StepsItem,
  StepsTrigger,
  StepsIndicator,
  StepsSeparator
} from "@/components/ui/steps";
```

```tsx
const items = [
  { title: 'Step 1' },
  { title: 'Step 2' },
  { title: 'Step 3' },
];

<StepsRoot count={items.length}>
  <StepsList>
    {items.map((item, index) => (
      <StepsItem key={index} index={index}>
        <StepsTrigger>
          <StepsIndicator>{index + 1}</StepsIndicator>
        </StepsTrigger>
        <StepsSeparator />
      </StepsItem>
    ))}
  </StepsList>
</StepsRoot>
```
