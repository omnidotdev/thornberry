### Examples

Let's see how to use the `NumberInput` components. Let's start with a basic example:

```tsx
import {
  NumberInputRoot,
  NumberInputControl,
  NumberInputInput,
  NumberInputIncrementTrigger,
  NumberInputDecrementTrigger
} from "@/components/ui/number-input"

export const BasicNumberInput = () => (
  <NumberInputRoot>
    <NumberInputControl>
      <NumberInputInput />
      <div className="flex flex-col border-l">
        <NumberInputIncrementTrigger />
        <NumberInputDecrementTrigger />
      </div>
    </NumberInputControl>
  </NumberInputRoot>
)
```

You can set minimum and maximum values for the number input:

```tsx
import {
  NumberInputRoot,
  NumberInputLabel,
  NumberInputControl,
  NumberInputInput,
  NumberInputIncrementTrigger,
  NumberInputDecrementTrigger
} from "@/components/ui/number-input"

export const RangeNumberInput = () => (
  <NumberInputRoot min={-10} max={40} defaultValue={22}>
    <NumberInputLabel>Temperature (°C)</NumberInputLabel>
    <NumberInputControl>
      <NumberInputInput />
      <div className="flex flex-col border-l">
        <NumberInputIncrementTrigger />
        <NumberInputDecrementTrigger />
      </div>
    </NumberInputControl>
  </NumberInputRoot>
)
```

You can customize the step size for increments and decrements:

```tsx
import {
  NumberInputRoot,
  NumberInputLabel,
  NumberInputControl,
  NumberInputInput,
  NumberInputIncrementTrigger,
  NumberInputDecrementTrigger
} from "@/components/ui/number-input"

export const StepNumberInput = () => (
  <NumberInputRoot min={0} max={100} step={5} defaultValue={50}>
    <NumberInputLabel>Volume</NumberInputLabel>
    <NumberInputControl>
      <NumberInputInput />
      <div className="flex flex-col border-l">
        <NumberInputIncrementTrigger />
        <NumberInputDecrementTrigger />
      </div>
    </NumberInputControl>
  </NumberInputRoot>
)
```
