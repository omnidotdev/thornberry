### Examples

Let's see how to use the `NumberInput` component. Let's start with a basic example:

```tsx
import { NumberInput } from "@/components/ui/number-input"

export const BasicNumberInput = () => (
  <NumberInput />
)
```

You can set minimum and maximum values for the number input:

```tsx
import { NumberInput } from "@/components/ui/number-input"

export const RangeNumberInput = () => (
  <NumberInput
    label="Temperature (°C)"
    min={-10}
    max={40}
    defaultValue={22}
  />
)
```

You can customize the step size for increments and decrements:

```tsx
import { NumberInput } from "@/components/ui/number-input"

export const StepNumberInput = () => (
  <NumberInput
    label="Volume"
    min={0}
    max={100}
    step={5}
    defaultValue={50}
  />
)
```

For more granular control, Thornberry provides NumberInput primitives:

```tsx
import {
  NumberInputRoot,
  NumberInputLabel,
  NumberInputControl,
  NumberInputInput,
  NumberInputIncrementTrigger,
  NumberInputDecrementTrigger
} from "@/components/ui/number-input"

export const NumberInputPrimitives = () => (
  <NumberInputRoot min={0} max={100} step={10} defaultValue={30}>
    <NumberInputLabel>Custom Number Input</NumberInputLabel>
    <NumberInputControl>
      <NumberInputInput />
      <div className="flex flex-col border-l justify-center h-full overflow-hidden">
        <NumberInputIncrementTrigger />
        <div className="h-px bg-border" />
        <NumberInputDecrementTrigger />
      </div>
    </NumberInputControl>
  </NumberInputRoot>
)
```
