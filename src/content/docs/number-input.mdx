---
title: Number Input
description: A form control that allows users to input numeric values with increment and decrement buttons
---

import {
  NumberInput,
  NumberInputContext,
  NumberInputControl,
  NumberInputDecrementTrigger,
  NumberInputIncrementTrigger,
  NumberInputInput,
  NumberInputLabel,
  NumberInputProvider,
  NumberInputRoot,
} from "@/registry/thornberry/components/number-input";

## Basic

```tsx
<NumberInput />
```

<NumberInput className="my-4" />

## Custom Range

```tsx
<NumberInput
  label="Temperature (°C)"
  min={-10}
  max={40}
  defaultValue={22}
/>
```

<NumberInput
  label="Temperature (°C)"
  min={-10}
  max={40}
  defaultValue={22}
  className="my-4"
/>

## Primitives

```tsx
<NumberInputRoot min={0} max={100} step={10} defaultValue={30}>
  <NumberInputLabel>Custom Number Input</NumberInputLabel>
  <NumberInputControl>
    <NumberInputInput />
    <div className="flex flex-col border-l justify-center h-full overflow-hidden">
      <NumberInputIncrementTrigger />
      <div className="h-px bg-border" />
      <NumberInputDecrementTrigger />
    </div>
  </NumberInputControl>
</NumberInputRoot>
```
