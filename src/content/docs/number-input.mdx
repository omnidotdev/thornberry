## Usage

```tsx
import {
  NumberInputRoot,
  NumberInputControl,
  NumberInputInput,
  NumberInputIncrementTrigger,
  NumberInputDecrementTrigger
} from "@/components/ui/number-input";
```

```tsx
<NumberInputRoot>
  <NumberInputControl>
    <NumberInputInput />
    <div className="flex flex-col h-10 border-l-0 border rounded-r-md">
      <NumberInputIncrementTrigger />
      <NumberInputDecrementTrigger />
    </div>
  </NumberInputControl>
</NumberInputRoot>
```
