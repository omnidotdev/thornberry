### Examples

Let's see how to use the `RatingGroup` components. Here's a basic example:

```tsx
import { RatingGroupRoot, RatingGroupLabel, RatingGroupControl, RatingGroupItem } from "@/components/ui/rating-group"

export const BasicRatingGroup = () => (
  <RatingGroupRoot>
    <RatingGroupLabel>Rate this product</RatingGroupLabel>
    <RatingGroupControl>
      <RatingGroupItem index={0} />
      <RatingGroupItem index={1} />
      <RatingGroupItem index={2} />
      <RatingGroupItem index={3} />
      <RatingGroupItem index={4} />
    </RatingGroupControl>
  </RatingGroupRoot>
)
```

You can specify a default rating:

```tsx
import { RatingGroupRoot, RatingGroupLabel, RatingGroupControl, RatingGroupItem } from "@/components/ui/rating-group"

export const DefaultRating = () => (
  <RatingGroupRoot defaultValue={3}>
    <RatingGroupLabel>Rate this product</RatingGroupLabel>
    <RatingGroupControl>
      <RatingGroupItem index={0} />
      <RatingGroupItem index={1} />
      <RatingGroupItem index={2} />
      <RatingGroupItem index={3} />
      <RatingGroupItem index={4} />
    </RatingGroupControl>
  </RatingGroupRoot>
)
```

You can customize the number of rating options:

```tsx
import { RatingGroupRoot, RatingGroupLabel, RatingGroupControl, RatingGroupItem } from "@/components/ui/rating-group"

export const TenStarRating = () => (
  <RatingGroupRoot>
    <RatingGroupLabel>Rate on a scale of 1-10</RatingGroupLabel>
    <RatingGroupControl>
      {Array.from({ length: 10 }).map((_, index) => (
        <RatingGroupItem key={index} index={index} />
      ))}
    </RatingGroupControl>
  </RatingGroupRoot>
)
```

```tsx
import {
  RatingGroupRoot,
  RatingGroupLabel,
  RatingGroupControl,
  RatingGroupItem
} from "@/components/ui/rating-group"
import { FiHeart } from "react-icons/fi"

export const CustomRatingGroup = () => (
  <RatingGroupRoot defaultValue={2}>
    <RatingGroupLabel>How much did you enjoy this article?</RatingGroupLabel>
    <RatingGroupControl>
      {Array.from({ length: 5 }).map((_, index) => (
        <RatingGroupItem 
          key={index} 
          index={index}
          className="data-checked:text-red-500 data-highlighted:text-red-300"
        >
          <FiHeart className="h-6 w-6 fill-current" />
        </RatingGroupItem>
      ))}
    </RatingGroupControl>
  </RatingGroupRoot>
)
```
