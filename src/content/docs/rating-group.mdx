## Usage

```tsx
import { RatingGroupRoot, RatingGroupLabel, RatingGroupControl, RatingGroupItem } from "@/components/ui/rating-group";
```

```tsx
<RatingGroupRoot>
  <RatingGroupLabel>Rate this product</RatingGroupLabel>
  <RatingGroupControl>
    <RatingGroupItem index={0} />
    <RatingGroupItem index={1} />
    <RatingGroupItem index={2} />
    <RatingGroupItem index={3} />
    <RatingGroupItem index={4} />
  </RatingGroupControl>
</RatingGroupRoot>
```
