## Usage

```tsx
import {
  MarqueeRoot,
  MarqueeViewport,
  MarqueeContent,
  MarqueeItem,
} from "@/components/ui/marquee";
```

```tsx
const items = ["Item 1", "Item 2", "Item 3", "Item 4"];

const Example = () => (
  <MarqueeRoot>
    <MarqueeViewport>
      <MarqueeContent>
        {items.map((item, i) => (
          <MarqueeItem key={i} className="px-8">
            {item}
          </MarqueeItem>
        ))}
      </MarqueeContent>
    </MarqueeViewport>
  </MarqueeRoot>
);
```
