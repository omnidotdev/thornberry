---
title: Checkbox
description: A control that allows the user to toggle between checked and not checked.
---

import { Check } from 'lucide-react';

import { CheckboxRoot, CheckboxHiddenInput, CheckboxControl, CheckboxIndicator, CheckboxLabel } from "@/registry/thornberry/components/checkbox";

<CheckboxRoot className="mb-8">
  <CheckboxHiddenInput />
  <CheckboxControl>
    <CheckboxIndicator>
      <Check />
    </CheckboxIndicator>
  </CheckboxControl>
  <CheckboxLabel>Checkbox</CheckboxLabel>
</CheckboxRoot>

<InstallationTabs filename="checkbox.json" />

## Usage

```tsx
import {
  CheckboxRoot,
  CheckboxHiddenInput,
  CheckboxControl,
  CheckboxIndicator,
  CheckboxLabel
} from "@/components/ui/checkbox";
```

```tsx
<CheckboxRoot>
  <CheckboxHiddenInput />
  <CheckboxControl>
    <CheckboxIndicator>
      <Check />
    </CheckboxIndicator>
  </CheckboxControl>
  <CheckboxLabel>Checkbox Label</CheckboxLabel>
</CheckboxRoot>
```
