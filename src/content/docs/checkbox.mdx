---
title: Checkbox
description: A control that allows the user to toggle between checked and not checked.
---

import {
  CheckboxControl,
  CheckboxHiddenInput,
  CheckboxIndicator,
  CheckboxLabel,
  CheckboxRoot,
} from "@/registry/thornberry/components/checkbox";
import { FiCheck, FiUser } from "react-icons/fi";

## Basic Checkbox

```tsx
<CheckboxRoot>
  <CheckboxHiddenInput />
  <CheckboxControl>
    <CheckboxIndicator>
      <FiCheck className="size-4" />
    </CheckboxIndicator>
  </CheckboxControl>
  <CheckboxLabel>Checkbox Label</CheckboxLabel>
</CheckboxRoot>
```

<CheckboxRoot className="mb-4">
  <CheckboxHiddenInput />
  <CheckboxControl>
    <CheckboxIndicator>
      <FiCheck className="size-4" />
    </CheckboxIndicator>
  </CheckboxControl>
  <CheckboxLabel>Checkbox Label</CheckboxLabel>
</CheckboxRoot>

## Custom Icon

```tsx
<CheckboxRoot defaultChecked>
  <CheckboxHiddenInput />
  <CheckboxControl>
    <CheckboxIndicator>
      <FiUser className="size-3" />
    </CheckboxIndicator>
  </CheckboxControl>
  <CheckboxLabel>Custom user icon</CheckboxLabel>
</CheckboxRoot>
```
