---

import { FiCheck } from "react-icons/fi";

import { CheckboxRoot, CheckboxHiddenInput, CheckboxControl, CheckboxIndicator, CheckboxLabel } from "@/registry/thornberry/components/checkbox";

<CheckboxRoot className="mb-8">
  <CheckboxHiddenInput />
  <CheckboxControl>
    <CheckboxIndicator>
      <FiCheck className="size-4" />
    </CheckboxIndicator>
  </CheckboxControl>
  <CheckboxLabel>Checkbox Label</CheckboxLabel>
</CheckboxRoot>

### Examples

Let's see how to use the `Checkbox` component. Let's start with a basic example:

```tsx
import { FiCheck } from "react-icons/fi"

import {
  CheckboxRoot,
  CheckboxHiddenInput,
  CheckboxControl,
  CheckboxIndicator,
  CheckboxLabel
} from "@/components/ui/checkbox"

export const BasicCheckbox = () => (
  <CheckboxRoot>
    <CheckboxHiddenInput />
    <CheckboxControl>
      <CheckboxIndicator>
        <FiCheck className="size-4" />
      </CheckboxIndicator>
    </CheckboxControl>
    <CheckboxLabel>Checkbox Label</CheckboxLabel>
  </CheckboxRoot>
)
```

You can customize the checkbox with different icons:

```tsx
import { FiUser } from "react-icons/fi"

import {
  CheckboxRoot,
  CheckboxHiddenInput,
  CheckboxControl,
  CheckboxIndicator,
  CheckboxLabel
} from "@/components/ui/checkbox"

export const CustomIconCheckbox = () => (
  <CheckboxRoot defaultChecked>
    <CheckboxHiddenInput />
    <CheckboxControl>
      <CheckboxIndicator>
        <FiUser className="size-3" />
      </CheckboxIndicator>
    </CheckboxControl>
    <CheckboxLabel>Custom user icon</CheckboxLabel>
  </CheckboxRoot>
)
```

You can create a checkbox group for multiple options:

```tsx
import { FiCheck } from "react-icons/fi"

import {
  CheckboxRoot,
  CheckboxHiddenInput,
  CheckboxControl,
  CheckboxIndicator,
  CheckboxLabel
} from "@/components/ui/checkbox"

export const CheckboxGroup = () => (
  <div className="flex flex-col gap-2">
    <CheckboxRoot name="options" value="option1">
      <CheckboxHiddenInput />
      <CheckboxControl>
        <CheckboxIndicator>
          <FiCheck className="size-4" />
        </CheckboxIndicator>
      </CheckboxControl>
      <CheckboxLabel>Option 1</CheckboxLabel>
    </CheckboxRoot>

    <CheckboxRoot name="options" value="option2">
      <CheckboxHiddenInput />
      <CheckboxControl>
        <CheckboxIndicator>
          <FiCheck className="size-4" />
        </CheckboxIndicator>
      </CheckboxControl>
      <CheckboxLabel>Option 2</CheckboxLabel>
    </CheckboxRoot>

    <CheckboxRoot name="options" value="option3" defaultChecked>
      <CheckboxHiddenInput />
      <CheckboxControl>
        <CheckboxIndicator>
          <FiCheck className="size-4" />
        </CheckboxIndicator>
      </CheckboxControl>
      <CheckboxLabel>Option 3</CheckboxLabel>
    </CheckboxRoot>
  </div>
)
```
