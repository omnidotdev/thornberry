## Usage

```tsx
import {
  JsonTreeViewRoot,
  JsonTreeViewTree,
} from "@/components/ui/json-tree-view";
```

```tsx
const Example = () => (
  <JsonTreeViewRoot
    data={{
      name: "Jane Smith",
      role: "Engineer",
      skills: ["React", "TypeScript", "Node.js"],
    }}
  >
    <JsonTreeViewTree />
  </JsonTreeViewRoot>
);
```
