---
title: Toggle Group
description: A set of two-state buttons that can be toggled on or off.
---

import { Bold, Italic, Underline } from "lucide-react";
import { ToggleGroupRoot, ToggleGroupItem } from "@/registry/thornberry/components/toggle-group";

<ToggleGroupRoot>
  <ToggleGroupItem value="bold">
    <Bold className="size-4" />
  </ToggleGroupItem>
  <ToggleGroupItem value="italic">
    <Italic className="size-4" />
  </ToggleGroupItem>
  <ToggleGroupItem value="underline">
    <Underline className="size-4" />
  </ToggleGroupItem>
</ToggleGroupRoot>

## Installation

<InstallationTabs filename="toggle-group.json" />

## Usage

```tsx
import { ToggleGroupRoot, ToggleGroupItem } from "@/components/ui/toggle-group";
```

```tsx
<ToggleGroupRoot>
  <ToggleGroupItem value="bold">
    <Bold className="size-4" />
  </ToggleGroupItem>
  <ToggleGroupItem value="italic">
    <Italic className="size-4" />
  </ToggleGroupItem>
</ToggleGroupRoot>
```
