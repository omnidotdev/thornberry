title: Alert
description: Displays a callout for user attention with status variants for different scenarios.
---

import { FiInfo, FiCheck } from "react-icons/fi";
import { IoWarningOutline } from "react-icons/io5";
import { BiErrorAlt } from "react-icons/bi";

import { AlertRoot, AlertIcon, AlertTitle, AlertDescription } from "@/registry/thornberry/components/alert";

<AlertRoot variant="info" className="mb-8">
  <AlertIcon>
    <FiInfo />
  </AlertIcon>
  <AlertTitle>Alert</AlertTitle>
  <AlertDescription>
    This is an alert component for important messages.
  </AlertDescription>
</AlertRoot>

## Installation

import { InstallationTabs } from "@/components";

<InstallationTabs filename="alert.json" />

## Usage

```tsx
import { AlertRoot, AlertIcon, AlertTitle, AlertDescription } from "@/components/ui/alert";
```

```tsx
<AlertRoot>
  <AlertIcon>
    <YourIcon />
  </AlertIcon>
  <AlertTitle>Title</AlertTitle>
  <AlertDescription>Description</AlertDescription>
</AlertRoot>
```
