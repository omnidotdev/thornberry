### Examples

Let's see how to use the `Alert` component. Let's start with a basic example:

```tsx
import { FiInfo } from "react-icons/fi"

import { AlertRoot, AlertIcon, AlertTitle, AlertDescription } from "@/components/ui/alert"

export const BasicAlert = () => (
  <AlertRoot>
    <AlertIcon>
      <FiInfo />
    </AlertIcon>

    <AlertTitle>Information</AlertTitle>

    <AlertDescription>
      This is an important information message for the user.
    </AlertDescription>
  </AlertRoot>
)
```

The `Alert` component come with different variants that you can use to represent different states:

```tsx
import { FiCheck, FiInfo } from "react-icons/fi"
import { IoWarningOutline } from "react-icons/io5"
import { BiErrorAlt } from "react-icons/bi"

import { AlertRoot, AlertIcon, AlertTitle, AlertDescription } from "@/components/ui/alert"

export const AlertVariants = () => (
  <>
    <AlertRoot>
      <AlertIcon>
        <FiInfo />
      </AlertIcon>

      <AlertTitle>Default</AlertTitle>

      <AlertDescription>
        For the best experience, update to the latest version.
      </AlertDescription>
    </AlertRoot>

    <AlertRoot variant="info">
      <AlertIcon>
        <FiInfo />
      </AlertIcon>

      <AlertTitle>Information</AlertTitle>

      <AlertDescription>
        Did you know platypuses are venomous?
      </AlertDescription>
    </AlertRoot>

    <AlertRoot variant="success">
      <AlertIcon>
        <FiCheck />
      </AlertIcon>

      <AlertTitle>Success</AlertTitle>

      <AlertDescription>
        Your profile has been updated successfully.
      </AlertDescription>
    </AlertRoot>

    <AlertRoot variant="warning">
      <AlertIcon>
        <IoWarningOutline />
      </AlertIcon>

      <AlertTitle>Warning</AlertTitle>

      <AlertDescription>
        Your storage is almost full. Please free up some space.
      </AlertDescription>
    </AlertRoot>

    <AlertRoot variant="error">
      <AlertIcon>
        <BiErrorAlt />
      </AlertIcon>

      <AlertTitle>Error</AlertTitle>

      <AlertDescription>
        There was a problem processing your request.
      </AlertDescription>
    </AlertRoot>
  </>
)
```
