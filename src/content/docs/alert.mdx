### Examples

Let's see how to use the `Alert` component. Let's start with a basic example:

```tsx
import { FiInfo } from "react-icons/fi"

import { Alert } from "@/components/ui/alert"

export const BasicAlert = () => (
  <Alert
    title="Information"
    description="This is an important information message for the user."
    icon={<FiInfo />}
  />
)
```

The Alert component comes with different variants that you can use to represent different states:

```tsx
import { FiCheck, FiInfo } from "react-icons/fi"
import { IoWarningOutline } from "react-icons/io5"
import { BiErrorAlt } from "react-icons/bi"

import { Alert } from "@/components/ui/alert"

export const AlertVariants = () => (
  <div className="flex flex-col gap-2">
    <Alert
      title="Default"
      description="For the best experience, update to the latest version."
      icon={<FiInfo />}
    />
    <Alert
      variant="info"
      title="Information"
      description="Did you know platypuses are venomous?"
      icon={<FiInfo />}
    />
    <Alert
      variant="success"
      title="Success"
      description="Your profile has been updated successfully."
      icon={<FiCheck />}
    />
    <Alert
      variant="warning"
      title="Warning"
      description="Your storage is almost full. Please free up some space."
      icon={<IoWarningOutline />}
    />
    <Alert
      variant="error"
      title="Error"
      description="There was a problem processing your request."
      icon={<BiErrorAlt />}
    />
  </div>
)
```

Thornberry also ships primitives for more granular control:

```tsx
import { FiInfo } from "react-icons/fi"

import { AlertRoot, AlertIcon, AlertTitle, AlertDescription } from "@/components/ui/alert"

export const AlertPrimitives = () => (
  <AlertRoot>
    <AlertIcon>
      <FiInfo />
    </AlertIcon>
    <AlertTitle>Custom Alert</AlertTitle>
    <AlertDescription>
      This alert was built using the primitive components for more flexibility.
    </AlertDescription>
  </AlertRoot>
)
```
