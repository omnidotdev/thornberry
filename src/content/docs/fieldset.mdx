## Usage

```tsx
import {
  FieldsetRoot,
  FieldsetLegend,
  FieldsetHelperText,
  FieldsetErrorText
} from "@/components/ui/fieldset";
```

```tsx
<FieldsetRoot>
  <FieldsetLegend>Legend</FieldsetLegend>
  <FieldsetHelperText>Helper text</FieldsetHelperText>
  {/* Fields */}
</FieldsetRoot>
```
