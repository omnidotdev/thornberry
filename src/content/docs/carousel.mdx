### Examples

Let's see how to use the `Carousel` components. Let's start with a basic example:

```tsx
import {
  CarouselRoot,
  CarouselItemGroup,
  CarouselItem,
  CarouselPrevTrigger,
  CarouselNextTrigger,
  CarouselIndicatorGroup,
  CarouselIndicator
} from "@/components/ui/carousel"

export const BasicCarousel = () => (
  <CarouselRoot slideCount={3} loop>
    <CarouselItemGroup>
      <CarouselItem index={0}>
        Slide 1
      </CarouselItem>

      <CarouselItem index={1}>
        Slide 2
      </CarouselItem>

      <CarouselItem index={2}>
        Slide 3
      </CarouselItem>
    </CarouselItemGroup>
    <CarouselPrevTrigger />

    <CarouselNextTrigger />

    <CarouselIndicatorGroup>
      <CarouselIndicator index={0} />

      <CarouselIndicator index={1} />

      <CarouselIndicator index={2} />
    </CarouselIndicatorGroup>
  </CarouselRoot>
)
```

You can hide the navigation controls:

```tsx
import {
  CarouselRoot,
  CarouselItemGroup,
  CarouselItem,
  CarouselIndicatorGroup,
  CarouselIndicator
} from "@/components/ui/carousel"

export const CarouselWithoutControls = () => (
  <CarouselRoot slideCount={3} loop>
    <CarouselItemGroup>
      <CarouselItem index={0}>
        Slide 1
      </CarouselItem>

      <CarouselItem index={1}>
        Slide 2
      </CarouselItem>

      <CarouselItem index={2}>
        Slide 3
      </CarouselItem>
    </CarouselItemGroup>

    <CarouselIndicatorGroup>
      <CarouselIndicator index={0} />
      
      <CarouselIndicator index={1} />
      
      <CarouselIndicator index={2} />

  </CarouselRoot>
)
```
