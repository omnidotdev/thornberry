### Examples

Let's see how to use the `Carousel` component. Let's start with a basic example:

```tsx
import { Carousel } from "@/components/ui/carousel"

export const BasicCarousel = () => (
  <Carousel
    items={[
      <div className="h-64 bg-slate-200 flex items-center justify-center rounded-md">
        <span className="text-2xl font-bold">Slide 1</span>
      </div>,
      <div className="h-64 bg-slate-300 flex items-center justify-center rounded-md">
        <span className="text-2xl font-bold">Slide 2</span>
      </div>,
      <div className="h-64 bg-slate-400 flex items-center justify-center rounded-md">
        <span className="text-2xl font-bold">Slide 3</span>
      </div>
    ]}
  />
)
```

You can hide the navigation controls:

```tsx
import { Carousel } from "@/components/ui/carousel"

export const CarouselWithoutControls = () => (
  <Carousel 
    showControls={false} 
    items={[
      <div className="h-64 bg-slate-200 flex items-center justify-center rounded-md">
        <span className="text-2xl font-bold">Slide 1</span>
      </div>,
      <div className="h-64 bg-slate-300 flex items-center justify-center rounded-md">
        <span className="text-2xl font-bold">Slide 2</span>
      </div>,
      <div className="h-64 bg-slate-400 flex items-center justify-center rounded-md">
        <span className="text-2xl font-bold">Slide 3</span>
      </div>
    ]} 
  />
)
```

You can hide the indicator dots:

```tsx
import { Carousel } from "@/components/ui/carousel"

export const CarouselWithoutIndicators = () => (
  <Carousel 
    showIndicators={false} 
    items={[
      <div className="h-64 bg-slate-200 flex items-center justify-center rounded-md">
        <span className="text-2xl font-bold">Slide 1</span>
      </div>,
      <div className="h-64 bg-slate-300 flex items-center justify-center rounded-md">
        <span className="text-2xl font-bold">Slide 2</span>
      </div>,
      <div className="h-64 bg-slate-400 flex items-center justify-center rounded-md">
        <span className="text-2xl font-bold">Slide 3</span>
      </div>
    ]} 
  />
)
```

For more granular control, Thornberry provides Carousel primitives:

```tsx
import {
  CarouselRoot,
  CarouselItemGroup,
  CarouselItem,
  CarouselPrevTrigger,
  CarouselNextTrigger,
  CarouselIndicatorGroup,
  CarouselIndicator
} from "@/components/ui/carousel"

export const CarouselPrimitives = () => (
  <CarouselRoot slideCount={3}>
    <CarouselItemGroup>
      <CarouselItem index={0}>
        <div className="h-64 bg-red-200 flex items-center justify-center rounded-md">
          <span className="text-2xl font-bold">Slide 1</span>
        </div>
      </CarouselItem>
      <CarouselItem index={1}>
        <div className="h-64 bg-red-300 flex items-center justify-center rounded-md">
          <span className="text-2xl font-bold">Slide 2</span>
        </div>
      </CarouselItem>
      <CarouselItem index={2}>
        <div className="h-64 bg-red-400 flex items-center justify-center rounded-md">
          <span className="text-2xl font-bold">Slide 3</span>
        </div>
      </CarouselItem>
    </CarouselItemGroup>
    <CarouselPrevTrigger />
    <CarouselNextTrigger />
    <CarouselIndicatorGroup>
      <CarouselIndicator index={0} />
      <CarouselIndicator index={1} />
      <CarouselIndicator index={2} />
    </CarouselIndicatorGroup>
  </CarouselRoot>
)
```
