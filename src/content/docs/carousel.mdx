## Usage

```tsx
import {
  CarouselRoot,
  CarouselItemGroup,
  CarouselItem,
  CarouselPrevTrigger,
  CarouselNextTrigger,
  CarouselIndicatorGroup,
  CarouselIndicator
} from "@/components/ui/carousel";
```

```tsx
<CarouselRoot slideCount={3} loop>
  <CarouselItemGroup>
    <CarouselItem index={0}>Slide 1</CarouselItem>
    <CarouselItem index={1}>Slide 2</CarouselItem>
    <CarouselItem index={2}>Slide 3</CarouselItem>
  </CarouselItemGroup>
  <CarouselPrevTrigger />
  <CarouselNextTrigger />
  <CarouselIndicatorGroup>
    <CarouselIndicator index={0} />
    <CarouselIndicator index={1} />
    <CarouselIndicator index={2} />
  </CarouselIndicatorGroup>
</CarouselRoot>
```
