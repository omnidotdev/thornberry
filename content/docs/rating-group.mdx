import { Star } from "lucide-react";
import { RatingGroupItem, RatingGroupControl, RatingGroupLabel, RatingGroupRoot } from "@/registry/thornberry/components/rating-group";

<RatingGroupRoot defaultValue={2} className="mb-8">
  <RatingGroupLabel>Rate this product</RatingGroupLabel>

      <RatingGroupItem key={index} index={index} >

## Usage

```tsx
import { Star } from "lucide-react";
import { RatingGroupRoot, RatingGroupLabel, RatingGroupControl, RatingGroupItem } from "@/components/ui/rating-group";
```

```tsx
<RatingGroupRoot defaultValue={2} className="mb-8">
  <RatingGroupLabel>Rate this product</RatingGroupLabel>
  <RatingGroupControl>
    {Array.from({ length: 5 }).map((_, index) => (
      <RatingGroupItem key={index} index={index} >
        <Star className="h-5 w-5 fill-current" />
      </RatingGroupItem>
    ))}
  </RatingGroupControl>
</RatingGroupRoot>
```

## Examples

#### Half Star
<Tabs items={["Preview", "Code"]} className="mt-4">
  <Tab value="Preview">
    <HalfStarRatingGroup />
  </Tab>
  <Tab value="Code">
    ```tsx
    import { StarHalfIcon, StarIcon } from "lucide-react";

    import {
      RatingGroupContext,
      RatingGroupControl,
      RatingGroupHiddenInput,
      RatingGroupItem,
      RatingGroupItemContext,
      RatingGroupLabel,
      RatingGroupRoot,
    } from "@/components/ui/rating-group";

   const Example = () => (
      <RatingGroupRoot count={5} defaultValue={3} allowHalf>
        <RatingGroupLabel>Half Star Example</RatingGroupLabel>
        <RatingGroupControl>
          <RatingGroupContext>
            {({ items }) =>
              items.map((item) => (
                <RatingGroupItem key={item} index={item}>
                  <RatingGroupItemContext>
                    {({ half, highlighted }) => {
                      if (half)
                        return <StarHalfIcon className="fill-amber-500" />;
                      if (highlighted)
                        return <StarIcon className="fill-amber-500" />;
                      return <StarIcon />;
                    }}
                  </RatingGroupItemContext>
                </RatingGroupItem>
              ))
            }
          </RatingGroupContext>
          <RatingGroupHiddenInput />
        </RatingGroupControl>
      </RatingGroupRoot>
    );
    ```
  </Tab>
</Tabs>
