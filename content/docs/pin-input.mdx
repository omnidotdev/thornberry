---
title: PIN Input
description: A component that allows users to input a sequence of characters, typically used for verification codes or passwords.
---

import { Shield } from 'lucide-react';
import { PinInputRoot, PinInputLabel, PinInputControl, PinInputInput } from "@/registry/thornberry/components/pin-input";

<PinInputRoot length={4} className="mb-8">
  <PinInputLabel className="text-sm font-medium text-gray-900 dark:text-gray-300">Verification Code</PinInputLabel>
  <PinInputControl>
    <PinInputInput index={0} />
    <PinInputInput index={1} />
    <PinInputInput index={2} />
    <PinInputInput index={3} />
  </PinInputControl>
</PinInputRoot>

## Usage

```tsx
import { PinInputRoot, PinInputLabel, PinInputControl, PinInputInput } from "@/components/ui/pin-input";
```

```tsx
<PinInputRoot length={4}>
  <PinInputLabel>Verification Code</PinInputLabel>
  <PinInputControl>
    <PinInputInput index={0} />
    <PinInputInput index={1} />
    <PinInputInput index={2} />
    <PinInputInput index={3} />
  </PinInputControl>
</PinInputRoot>
```
