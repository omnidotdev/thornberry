---
import { User } from 'lucide-react';

import { AvatarRoot, AvatarFallback, AvatarImage } from "@/registry/thornberry/components/avatar";

<AvatarRoot className="mb-8">
  <AvatarImage src="/img/elon-musk.jpg" alt="Elon Musk" />
  <AvatarFallback>EM</AvatarFallback>
</AvatarRoot>

## Installation

<InstallationTabs filename="avatar.json" />

## Usage

```tsx
import { AvatarRoot, AvatarFallback, AvatarImage } from "@/components/ui/avatar";
```

```tsx
<AvatarRoot>
  <AvatarImage src="/path-to-image.jpg" alt="User name" />
  <AvatarFallback>UN</AvatarFallback>
</AvatarRoot>
```
