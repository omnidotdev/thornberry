---
title: Hover Card
description: A card that appears when hovering over a trigger element, providing additional information.
---

import { User, Mail, MapPin } from "lucide-react";
import { Badge } from "@/registry/thornberry/components/badge";
import { HoverCardRoot, HoverCardTrigger, HoverCardContent, HoverCardPositioner } from "@/registry/thornberry/components/hover-card";

<div className="bg-accent w-full h-48 mb-8 rounded-md flex items-center justify-center">
  <HoverCardRoot>
    <HoverCardTrigger className="hover:underline h-8 flex items-center px-3 py-2">
      Hover me
    </HoverCardTrigger>
    <HoverCardPositioner>
      <HoverCardContent>
        <p>Here is the hover card content</p>
      </HoverCardContent>
    </HoverCardPositioner>
  </HoverCardRoot>
</div>

## Usage

```tsx
import {
  HoverCardRoot,
  HoverCardTrigger,
  HoverCardContent,
  HoverCardPositioner,
} from "@/components/ui/hover-card";
```

```tsx
<div className="bg-accent w-full h-48 mb-8 rounded-md flex items-center justify-center">
  <HoverCardRoot>
    <HoverCardTrigger className="hover:underline h-8 flex items-center px-3 py-2">
      Hover me
    </HoverCardTrigger>
    <HoverCardPositioner>
      <HoverCardContent>
        <p>Here is the hover card content</p>
      </HoverCardContent>
    </HoverCardPositioner>
  </HoverCardRoot>
</div>
```
